8 DYN_FUNC void jacobiRotate(SquareMatrix<Real, 3> &A, SquareMatrix<Real, 3> &R, int p, int q)
9 {
10 // rotates A through phi in pq-plane to set A(p,q) = 0
11 // rotation stored in R whose columns are eigenvectors of A
12 if (A(p, q) == 0.0f)
13 return;
14
15 Real d = (A(p, p) - A(q, q)) / (2.0f*A(p, q));
16 Real t = 1.0f / (fabs(d) + sqrt(d*d + 1.0f));
17 if (d < 0.0f) t = -t;
18 Real c = 1.0f / sqrt(t*t + 1);
19 Real s = t*c;
20 A(p, p) += t*A(p, q);
21 A(q, q) -= t*A(p, q);
22 A(p, q) = A(q, p) = 0.0f;
23 // transform A
24 int k;
25 for (k = 0; k < 3; k++) {
26 if (k != p && k != q) {
27 Real Akp = c*A(k, p) + s*A(k, q);
28 Real Akq = -s*A(k, p) + c*A(k, q);
29 A(k, p) = A(p, k) = Akp;
30 A(k, q) = A(q, k) = Akq;
31 }
32 }
33 // store rotation in R
34 for (k = 0; k < 3; k++) {
35 Real Rkp = c*R(k, p) + s*R(k, q);
36 Real Rkq = -s*R(k, p) + c*R(k, q);
37 R(k, p) = Rkp;
38 R(k, q) = Rkq;
39 }
40 }
