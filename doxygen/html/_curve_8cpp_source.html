318 void Curve::rebuildHandlePoint(std::vector<Coord2D> coordSet)
319 {
320 myHandlePoint.clear();
321 int ptnum = coordSet.size();
322 for (size_t i = 0;i < ptnum; i++ )
323 {
324 dyno::Vec2f P(coordSet[i].x, coordSet[i].y);
325 dyno::Vec2f p1;
326 dyno::Vec2f p2;
327 dyno::Vec2f N;
328
329 int id = i;
330 int f;
331 int s;
332 if (ptnum < 2)
333 {
334 N = Vec2f(1, 0);
335 }
336 else
337 {
338 f = id - 1;
339 s = id + 1;
340 if (id == 0)//首点
341 {
342 N[0] = coordSet[s].x - coordSet[id].x;
343 N[1] = coordSet[s].y - coordSet[id].y;
344 }
345 else if (id == ptnum - 1)//末点
346 {
347 N[0] = coordSet[id].x - coordSet[f].x;
348 N[1] = coordSet[id].y - coordSet[f].y;
349 }
350 else//中间点
351 {
352 N[0] = coordSet[s].x - coordSet[f].x;
353 N[1] = coordSet[s].y - coordSet[f].y;
354 }
355 }
356
357 N.normalize();
358 double length = 0.05;
359 p1 = P - N * length;
360 p2 = P + N * length;
361
362 myHandlePoint.push_back(Coord2D(p1));
363 myHandlePoint.push_back(Coord2D(p2));

369 void Curve::updateBezierPointToBezierSet(Coord2D p0, Coord2D p1, Coord2D p2, Coord2D p3, std::vector<Coord2D>& bezierSet)
370 {
371 Coord2D P[4] = { p0,p1,p2,p3 };
372
373 int n = 4;
374 Coord2D Pf;
375 double t;
376 double unit = 1 / segment;
377
378 for (t = 0; t < 1 ; t += unit)
379 {
380 P[0] = p0;
381 P[1] = p1;
382 P[2] = p2;
383 P[3] = p3;
384
385 int x = n;
386 while (1)
387 {
388 if (x == 1)
389 break;
390 for (int i = 0; i < x - 1; i++)
391 {
392 Pf.x = (P[i + 1].x - P[i].x) * t + P[i].x;
393 Pf.y = (P[i + 1].y - P[i].y) * t + P[i].y;
394 P[i] = Pf;
395
396 }
397 x--;
398 }
399
400 bezierSet.push_back(Pf);
