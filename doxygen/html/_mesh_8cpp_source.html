42 { // create sphere
43 // http://www.songho.ca/opengl/gl_sphere.html

47 std::vector<float> texCoords;
48 std::vector<int> indices;
49
50 float x, y, z, xy; // vertex position
51 float nx, ny, nz, lengthInv = 1.0f / radius; // vertex normal
52 float s, t; // vertex texCoord
53
54 const float PI = 3.14159265f;
55 float sectorStep = 2 * PI / sectorCount;
56 float stackStep = PI / stackCount;
57 float sectorAngle, stackAngle;
58
59 for (int i = 0; i <= stackCount; ++i)
60 {
61 stackAngle = PI / 2 - i * stackStep; // starting from pi/2 to -pi/2
62 xy = radius * cosf(stackAngle); // r * cos(u)
63 z = radius * sinf(stackAngle); // r * sin(u)
64
65 // add (sectorCount+1) vertices per stack
66 // the first and last vertices have same position and normal, but different tex coords
67 for (int j = 0; j <= sectorCount; ++j)
68 {
69 sectorAngle = j * sectorStep; // starting from 0 to 2pi
70
71 // vertex position (x, y, z)
72 x = xy * cosf(sectorAngle); // r * cos(u) * cos(v)
73 y = xy * sinf(sectorAngle); // r * cos(u) * sin(v)
74 vertices.push_back(x);
75 vertices.push_back(y);
76 vertices.push_back(z);
77
78 // normalized vertex normal (nx, ny, nz)
79 nx = x * lengthInv;
80 ny = y * lengthInv;
81 nz = z * lengthInv;
82 normals.push_back(nx);
83 normals.push_back(ny);
84 normals.push_back(nz);
85
86 // vertex tex coord (s, t) range between [0, 1]
87 s = (float)j / sectorCount;
88 t = (float)i / stackCount;
89 texCoords.push_back(s);
90 texCoords.push_back(t);
91 }
92 }
93
94 // generate CCW index list of sphere triangles
95 int k1, k2;
96 for (int i = 0; i < stackCount; ++i)
97 {
98 k1 = i * (sectorCount + 1); // beginning of current stack
99 k2 = k1 + sectorCount + 1; // beginning of next stack
100
101 for (int j = 0; j < sectorCount; ++j, ++k1, ++k2)
102 {
103 // 2 triangles per sector excluding first and last stacks
104 // k1 => k2 => k1+1
105 if (i != 0)
106 {
107 indices.push_back(k1);
108 indices.push_back(k2);
109 indices.push_back(k1 + 1);
110 }
111
112 // k1+1 => k2 => k2+1
113 if (i != (stackCount - 1))
114 {
115 indices.push_back(k1 + 1);
116 indices.push_back(k2);
117 indices.push_back(k2 + 1);
118 }
119 }
120 }
