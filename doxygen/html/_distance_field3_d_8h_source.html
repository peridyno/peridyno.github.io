139 GPU_FUNC void DistanceField3D<TDataType>::getDistance(const Coord &p, Real &d, Coord &normal)
140 {
141 // get cell and lerp values
142 Coord fp = (p - m_left)*Coord(1.0 / m_h[0], 1.0 / m_h[1], 1.0 / m_h[2]);
143 const int i = (int)floor(fp[0]);
144 const int j = (int)floor(fp[1]);
145 const int k = (int)floor(fp[2]);
146 if (i < 0 || i >= m_distance.nx() - 1 || j < 0 || j >= m_distance.ny() - 1 || k < 0 || k >= m_distance.nz() - 1) {
147 if (m_bInverted) d = -100000.0f;
148 else d = 100000.0f;
149 normal = Coord(0);
150 return;
151 }
152 Coord ip = Coord(i, j, k);
153
154 Coord alphav = fp - ip;
155 Real alpha = alphav[0];
156 Real beta = alphav[1];
157 Real gamma = alphav[2];
158
159 Real d000 = m_distance(i, j, k);
160 Real d100 = m_distance(i + 1, j, k);
161 Real d010 = m_distance(i, j + 1, k);
162 Real d110 = m_distance(i + 1, j + 1, k);
163 Real d001 = m_distance(i, j, k + 1);
164 Real d101 = m_distance(i + 1, j, k + 1);
165 Real d011 = m_distance(i, j + 1, k + 1);
166 Real d111 = m_distance(i + 1, j + 1, k + 1);
167
168 Real dx00 = lerp(d000, d100, alpha);
169 Real dx10 = lerp(d010, d110, alpha);
170 Real dxy0 = lerp(dx00, dx10, beta);
171
172 Real dx01 = lerp(d001, d101, alpha);
173 Real dx11 = lerp(d011, d111, alpha);
174 Real dxy1 = lerp(dx01, dx11, beta);
175
176 Real d0y0 = lerp(d000, d010, beta);
177 Real d0y1 = lerp(d001, d011, beta);
178 Real d0yz = lerp(d0y0, d0y1, gamma);
179
180 Real d1y0 = lerp(d100, d110, beta);
181 Real d1y1 = lerp(d101, d111, beta);
182 Real d1yz = lerp(d1y0, d1y1, gamma);
183
184 Real dx0z = lerp(dx00, dx01, gamma);
185 Real dx1z = lerp(dx10, dx11, gamma);
186
187 normal[0] = d0yz - d1yz;
188 normal[1] = dx0z - dx1z;
189 normal[2] = dxy0 - dxy1;
190
191 Real l = normal.norm();
192 if (l < 0.0001f) normal = Coord(0);
193 else normal = normal.normalize();
194
195 d = (1.0f - gamma) * dxy0 + gamma * dxy1;
196 }
