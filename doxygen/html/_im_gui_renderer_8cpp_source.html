206bool ImGuiRenderer::createFontsTexture()
207{
208 // Select current context
209 ImGui::SetCurrentContext(g_ctx);
210
211 // Build texture atlas
212 ImGuiIO& io = ImGui::GetIO();
213 unsigned char* pixels;
214 int width, height;
215 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height); // Load as RGBA 32-bits (75% of the memory is wasted, but default font is so small) because it is more likely to be compatible with user's existing shaders. If your ImTextureId represent a higher-level concept than just a GL texture id, consider calling GetTexDataAsAlpha8() instead to save on GPU memory.
216
217 // Upload texture to graphics system
218 GLint last_texture;
219 glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
220 glGenTextures(1, &g_FontTexture);
221 glBindTexture(GL_TEXTURE_2D, g_FontTexture);
222 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
223 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
224 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, pixels);
225
226 // Store our identifier
227 io.Fonts->TexID = (void *)(size_t)g_FontTexture;
228
229 // Restore state
230 glBindTexture(GL_TEXTURE_2D, last_texture);
231
232 return true;
233}

235bool ImGuiRenderer::createDeviceObjects()
236{
237 // Select current context
238 ImGui::SetCurrentContext(g_ctx);
239
240 // Backup GL state
241 GLint last_texture, last_array_buffer, last_vertex_array;
242 glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
243 glGetIntegerv(GL_ARRAY_BUFFER_BINDING, &last_array_buffer);
244 glGetIntegerv(GL_VERTEX_ARRAY_BINDING, &last_vertex_array);
245
246 const GLchar *vertex_shader =
247 IMGUIRENDERER_GLSL_VERSION
248 "uniform mat4 ProjMtx;\n"
249 "in vec2 Position;\n"
250 "in vec2 UV;\n"
251 "in vec4 Color;\n"
252 "out vec2 Frag_UV;\n"
253 "out vec4 Frag_Color;\n"
254 "void main()\n"
255 "{\n"
256 " Frag_UV = UV;\n"
257 " Frag_Color = Color;\n"
258 " gl_Position = ProjMtx * vec4(Position.xy,0,1);\n"
259 "}\n";
260
261 const GLchar* fragment_shader =
262 IMGUIRENDERER_GLSL_VERSION
263 "precision mediump float;"
264 "uniform sampler2D Texture;\n"
265 "in vec2 Frag_UV;\n"
266 "in vec4 Frag_Color;\n"
267 "out vec4 Out_Color;\n"
268 "void main()\n"
269 "{\n"
270 " Out_Color = Frag_Color * texture( Texture, Frag_UV.st);\n"
271 "}\n";
272
273 g_ShaderHandle = glCreateProgram();
274 g_VertHandle = glCreateShader(GL_VERTEX_SHADER);
275 g_FragHandle = glCreateShader(GL_FRAGMENT_SHADER);
276 glShaderSource(g_VertHandle, 1, &vertex_shader, 0);
277 glShaderSource(g_FragHandle, 1, &fragment_shader, 0);
278 glCompileShader(g_VertHandle);
279 glCompileShader(g_FragHandle);
280 glAttachShader(g_ShaderHandle, g_VertHandle);
281 glAttachShader(g_ShaderHandle, g_FragHandle);
282 glLinkProgram(g_ShaderHandle);
283
284 g_AttribLocationTex = glGetUniformLocation(g_ShaderHandle, "Texture");
285 g_AttribLocationProjMtx = glGetUniformLocation(g_ShaderHandle, "ProjMtx");
286 g_AttribLocationPosition = glGetAttribLocation(g_ShaderHandle, "Position");
287 g_AttribLocationUV = glGetAttribLocation(g_ShaderHandle, "UV");
288 g_AttribLocationColor = glGetAttribLocation(g_ShaderHandle, "Color");
289
290 glGenBuffers(1, &g_VboHandle);
291 glGenBuffers(1, &g_ElementsHandle);
292
293 glGenVertexArrays(1, &g_VaoHandle);
294 glBindVertexArray(g_VaoHandle);
295 glBindBuffer(GL_ARRAY_BUFFER, g_VboHandle);
296 glEnableVertexAttribArray(g_AttribLocationPosition);
297 glEnableVertexAttribArray(g_AttribLocationUV);
298 glEnableVertexAttribArray(g_AttribLocationColor);
299
300#define OFFSETOF(TYPE, ELEMENT) ((size_t)&(((TYPE *)0)->ELEMENT))
301 glVertexAttribPointer(g_AttribLocationPosition, 2, GL_FLOAT, GL_FALSE, sizeof(ImDrawVert), (GLvoid*)OFFSETOF(ImDrawVert, pos));
302 glVertexAttribPointer(g_AttribLocationUV, 2, GL_FLOAT, GL_FALSE, sizeof(ImDrawVert), (GLvoid*)OFFSETOF(ImDrawVert, uv));
303 glVertexAttribPointer(g_AttribLocationColor, 4, GL_UNSIGNED_BYTE, GL_TRUE, sizeof(ImDrawVert), (GLvoid*)OFFSETOF(ImDrawVert, col));
304#undef OFFSETOF
305
306 createFontsTexture();
307
308 // Restore modified GL state
309 glBindTexture(GL_TEXTURE_2D, last_texture);
310 glBindBuffer(GL_ARRAY_BUFFER, last_array_buffer);
311 glBindVertexArray(last_vertex_array);
312
313 return true;
314}

316void ImGuiRenderer::newFrame()
317{
318 // Select current context
319 ImGui::SetCurrentContext(g_ctx);
320
321 if (!g_FontTexture)
322 createDeviceObjects();
323
324 ImGuiIO& io = ImGui::GetIO();
325
326 // Setup display size (every frame to accommodate for window resizing)
327 io.DisplaySize = ImVec2(m_window->size().width(), m_window->size().height());
328 io.DisplayFramebufferScale = ImVec2(m_window->devicePixelRatio(), m_window->devicePixelRatio());
329
330 // Setup time step
331 double current_time = QDateTime::currentMSecsSinceEpoch() / double(1000);
332 io.DeltaTime = g_Time > 0.0 ? (float)(current_time - g_Time) : (float)(1.0f/60.0f);
333 io.DeltaTime = io.DeltaTime < 0.001f ? 0.001f : io.DeltaTime;
334 g_Time = current_time;
335
336
337 // If ImGui wants to set cursor position (for example, during navigation by using keyboard)
338 // we need to do it here (before getting `QCursor::pos()` below).
339 setCursorPos(io);
340
341 // Setup inputs
342 // (we already got mouse wheel, keyboard keys & characters from glfw callbacks polled in glfwPollEvents())
343 if (m_window->isActive())
344 {
345 const QPoint pos = m_window->mapFromGlobal(QCursor::pos());
346 io.MousePos = ImVec2(pos.x(), pos.y()); // Mouse position in screen coordinates (set to -1,-1 if no mouse / on another screen, etc.)
347 }
348 else
349 {
350 io.MousePos = ImVec2(-1,-1);
351 }
352
353 for (int i = 0; i < 3; i++)
354 {
355 io.MouseDown[i] = g_MousePressed[i];
356 }
357
358 io.MouseWheelH = g_MouseWheelH;
359 io.MouseWheel = g_MouseWheel;
360 g_MouseWheelH = 0;
361 g_MouseWheel = 0;
362
363
364 updateCursorShape(io);
365
366
367 // Start the frame
368 ImGui::NewFrame();
369}

398void ImGuiRenderer::onWheel(QWheelEvent *event)
399{
400 // Select current context
401 ImGui::SetCurrentContext(g_ctx);
402
403 // Handle horizontal component
404 if(event->pixelDelta().x() != 0)
405 {
406 g_MouseWheelH += event->pixelDelta().x() / (ImGui::GetTextLineHeight());
407 } else {
408 // Magic number of 120 comes from Qt doc on QWheelEvent::pixelDelta()
409 g_MouseWheelH += event->angleDelta().x() / 120;
410 }
411

417 } else {

423void ImGuiRenderer::onKeyPressRelease(QKeyEvent *event)
424{
425 // Select current context
426 ImGui::SetCurrentContext(g_ctx);
427
428 ImGuiIO& io = ImGui::GetIO();
429
430 const bool key_pressed = (event->type() == QEvent::KeyPress);
431
432 // Translate `Qt::Key` into `ImGuiKey`, and apply 'pressed' state for that key
433 const auto key_it = keyMap.constFind( event->key() );
434 if (key_it != keyMap.constEnd()) { // Qt's key found in keyMap
435 const int imgui_key = *(key_it);
436 io.KeysDown[imgui_key] = key_pressed;
437 }
438
439 if (key_pressed) {
440 const QString text = event->text();
441 if (text.size() == 1) {
442 io.AddInputCharacter( text.at(0).unicode() );
443 }
444 }
445
446#ifdef Q_OS_MAC
447 io.KeyCtrl = event->modifiers() & Qt::MetaModifier;
448 io.KeyShift = event->modifiers() & Qt::ShiftModifier;
449 io.KeyAlt = event->modifiers() & Qt::AltModifier;
450 io.KeySuper = event->modifiers() & Qt::ControlModifier; // Comamnd key
451#else
452 io.KeyCtrl = event->modifiers() & Qt::ControlModifier;
453 io.KeyShift = event->modifiers() & Qt::ShiftModifier;
454 io.KeyAlt = event->modifiers() & Qt::AltModifier;
455 io.KeySuper = event->modifiers() & Qt::MetaModifier;

459void ImGuiRenderer::updateCursorShape(const ImGuiIO& io)
460{
461 // NOTE: This code will be executed, only if the following flags have been set:
462 // - backend flag: `ImGuiBackendFlags_HasMouseCursors` - enabled
463 // - config flag: `ImGuiConfigFlags_NoMouseCursorChange` - disabled
464
465#ifndef QT_NO_CURSOR
466 if (io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange)
467 return;
468
469 const ImGuiMouseCursor imgui_cursor = ImGui::GetMouseCursor();
470 if (io.MouseDrawCursor || (imgui_cursor == ImGuiMouseCursor_None))
471 {
472 // Hide OS mouse cursor if imgui is drawing it or if it wants no cursor
473 m_window->setCursorShape(Qt::CursorShape::BlankCursor);
474 }
475 else
476 {
477 // Show OS mouse cursor
478
479 // Translate `ImGuiMouseCursor` into `Qt::CursorShape` and show it, if we can
480 const auto cursor_it = cursorMap.constFind( imgui_cursor );
481 if(cursor_it != cursorMap.constEnd()) // `Qt::CursorShape` found for `ImGuiMouseCursor`
482 {
483 const Qt::CursorShape qt_cursor_shape = *(cursor_it);
484 m_window->setCursorShape(qt_cursor_shape);
485 }
486 else // shape NOT found - use default
487 {
488 m_window->setCursorShape(Qt::CursorShape::ArrowCursor);
489 }
490 }
491#else
492 Q_UNUSED(io);
493#endif
494}

511bool ImGuiRenderer::eventFilter(QObject *watched, QEvent *event)
512{
513 if (watched == m_window->object()) {
514 switch (event->type()) {
515 case QEvent::MouseButtonDblClick:
516 case QEvent::MouseButtonPress:
517 case QEvent::MouseButtonRelease:
518 this->onMousePressedChange(static_cast<QMouseEvent*>(event));
519 break;
520 case QEvent::Wheel:
521 this->onWheel(static_cast<QWheelEvent*>(event));
522 break;
523 case QEvent::KeyPress:
524 case QEvent::KeyRelease:
525 this->onKeyPressRelease(static_cast<QKeyEvent*>(event));
526 break;
527 default:
528 break;
529 }
530 }
531 return QObject::eventFilter(watched, event);
532}
