264 void PMainWindow::showModuleEditor(Qt::QtNode& s)
265 {
266 Qt::QtNodeWidget* clickedNode = nullptr;
267
268 clickedNode = dynamic_cast<Qt::QtNodeWidget*>(s.nodeDataModel());
269
270 if (clickedNode == nullptr)
271 return;
272
273 QString caption = s.nodeDataModel()->caption();
274
275 PModuleEditor* moduleEditor = new PModuleEditor(clickedNode);
276 moduleEditor->setWindowTitle("Module Editor -- " + caption);
277 moduleEditor->resize(1024, 600);
278 moduleEditor->setMinimumSize(512, 360);
279
280 moduleEditor->setWindowModality(Qt::WindowModal);
281 moduleEditor->setAttribute(Qt::WA_ShowModal, true);
282 moduleEditor->setAttribute(Qt::WA_DeleteOnClose, true);
283 moduleEditor->show();
284
285 connect(moduleEditor, &PModuleEditor::changed, mOpenGLWidget, &POpenGLWidget::updateGraphicsContext);
286 connect(moduleEditor->moduleFlowScene(), &Qt::QtModuleFlowScene::nodeExportChanged, mNodeFlowView->flowScene(), &Qt::QtNodeFlowScene::updateNodeGraphView);
