263 void PMainWindow::showModuleEditor(Qt::QtNode& s)
264 {
265 Qt::QtNodeWidget* clickedNode = nullptr;
266
267 clickedNode = dynamic_cast<Qt::QtNodeWidget*>(s.nodeDataModel());
268
269 if (clickedNode == nullptr)
270 return;
271
272 QString caption = s.nodeDataModel()->caption();
273
274 PModuleEditor* moduleEditor = new PModuleEditor(clickedNode);
275 moduleEditor->setWindowTitle("Module Editor -- " + caption);
276 moduleEditor->resize(1024, 600);
277 moduleEditor->setMinimumSize(512, 360);
278
279 moduleEditor->setWindowModality(Qt::WindowModal);
280 moduleEditor->setAttribute(Qt::WA_ShowModal, true);
281 moduleEditor->setAttribute(Qt::WA_DeleteOnClose, true);
282 moduleEditor->show();
283
284 connect(moduleEditor, &PModuleEditor::changed, mOpenGLWidget, &POpenGLWidget::updateGraphicsContext);
285 connect(moduleEditor->moduleFlowScene(), &Qt::QtModuleFlowScene::nodeExportChanged, mNodeFlowView->flowScene(), &Qt::QtNodeFlowScene::updateNodeGraphView);
286 }
