319 void PMainWindow::showModuleEditor()
320 {
321 auto nodes = mNodeFlowView->node_scene->selectedNodes();
322 Qt::QtNodeWidget* clickedNode = nullptr;
323 if (nodes.size() > 0) {
324 clickedNode = dynamic_cast<Qt::QtNodeWidget*>(nodes[0]->nodeDataModel());
325 }
326
327 if (clickedNode == nullptr)
328 return;
329
330 PModuleEditor* moduelEditor = new PModuleEditor(clickedNode);
331 moduelEditor->setWindowTitle("Module Flow Editor");
332 moduelEditor->resize(1024, 600);
333 moduelEditor->setMinimumSize(512, 360);
334
335 moduelEditor->setWindowModality(Qt::ApplicationModal);
336 moduelEditor->setAttribute(Qt::WA_ShowModal, true);
337 moduelEditor->setAttribute(Qt::WA_DeleteOnClose, true);
338 moduelEditor->show();
339
340 connect(moduelEditor, &PModuleEditor::changed, mOpenGLWidget, &POpenGLWidget::updateGraphicsContext);
341 connect(moduelEditor->moduleFlowScene(), &Qt::QtModuleFlowScene::nodeExportChanged, mNodeFlowView->node_scene, &Qt::QtNodeFlowScene::updateNodeGraphView);
342 }
