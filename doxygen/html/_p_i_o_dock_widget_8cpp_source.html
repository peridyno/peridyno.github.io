94 void PIODockWidget::updateContextMenu()
95 {
96 const Qt::DockWidgetArea area = mainWindow->dockWidgetArea(this);
97 const Qt::DockWidgetAreas areas = allowedAreas();
98
99 closableAction->setChecked(features() & QDockWidget::DockWidgetClosable);
100 if (windowType() == Qt::Drawer) {
101 floatableAction->setEnabled(false);
102 floatingAction->setEnabled(false);
103 movableAction->setEnabled(false);
104 verticalTitleBarAction->setChecked(false);
105 }
106 else {
107 floatableAction->setChecked(features() & QDockWidget::DockWidgetFloatable);
108 floatingAction->setChecked(isWindow());
109 // done after floating, to get 'floatable' correctly initialized
110 movableAction->setChecked(features() & QDockWidget::DockWidgetMovable);
111 verticalTitleBarAction
112 ->setChecked(features() & QDockWidget::DockWidgetVerticalTitleBar);
113 }
114
115 allowLeftAction->setChecked(isAreaAllowed(Qt::LeftDockWidgetArea));
116 allowRightAction->setChecked(isAreaAllowed(Qt::RightDockWidgetArea));
117 allowTopAction->setChecked(isAreaAllowed(Qt::TopDockWidgetArea));
118 allowBottomAction->setChecked(isAreaAllowed(Qt::BottomDockWidgetArea));
119
120 if (allowedAreasActions->isEnabled()) {
121 allowLeftAction->setEnabled(area != Qt::LeftDockWidgetArea);
122 allowRightAction->setEnabled(area != Qt::RightDockWidgetArea);
123 allowTopAction->setEnabled(area != Qt::TopDockWidgetArea);
124 allowBottomAction->setEnabled(area != Qt::BottomDockWidgetArea);
125 }
126
127 {
128 const QSignalBlocker blocker(leftAction);
129 leftAction->setChecked(area == Qt::LeftDockWidgetArea);
130 }
131 {
132 const QSignalBlocker blocker(rightAction);
133 rightAction->setChecked(area == Qt::RightDockWidgetArea);
134 }
135 {
136 const QSignalBlocker blocker(topAction);
137 topAction->setChecked(area == Qt::TopDockWidgetArea);
138 }
139 {
140 const QSignalBlocker blocker(bottomAction);
141 bottomAction->setChecked(area == Qt::BottomDockWidgetArea);
142 }
143
144 if (areaActions->isEnabled()) {
145 leftAction->setEnabled(areas & Qt::LeftDockWidgetArea);
146 rightAction->setEnabled(areas & Qt::RightDockWidgetArea);
147 topAction->setEnabled(areas & Qt::TopDockWidgetArea);
148 bottomAction->setEnabled(areas & Qt::BottomDockWidgetArea);
149 }
150
151 tabMenu->clear();
152 splitHMenu->clear();
153 splitVMenu->clear();
154 QList<PIODockWidget*> dock_list = mainWindow->findChildren<PIODockWidget*>();
155 foreach(PIODockWidget *dock, dock_list) {
156 tabMenu->addAction(dock->objectName());
157 splitHMenu->addAction(dock->objectName());
158 splitVMenu->addAction(dock->objectName());
159 }
160 }

162 static PIODockWidget *findByName(const QMainWindow *mainWindow, const QString &name)
163 {
164 foreach(PIODockWidget *dock, mainWindow->findChildren<PIODockWidget*>()) {
165 if (name == dock->objectName())
166 return dock;
167 }
168 return Q_NULLPTR;
169 }

171 void PIODockWidget::splitInto(QAction *action)
172 {
173 PIODockWidget *target = findByName(mainWindow, action->text());
174 if (!target)
175 return;
176
177 const Qt::Orientation o = action->parent() == splitHMenu
178 ? Qt::Horizontal : Qt::Vertical;
179 mainWindow->splitDockWidget(target, this, o);
180 }

182 void PIODockWidget::tabInto(QAction *action)
183 {
184 if (PIODockWidget *target = findByName(mainWindow, action->text()))
185 mainWindow->tabifyDockWidget(target, this);
186 }

208 void PIODockWidget::allow(Qt::DockWidgetArea area, bool a)
209 {
210 Qt::DockWidgetAreas areas = allowedAreas();
211 areas = a ? areas | area : areas & ~area;
212 setAllowedAreas(areas);
213
214 if (areaActions->isEnabled()) {
215 leftAction->setEnabled(areas & Qt::LeftDockWidgetArea);
216 rightAction->setEnabled(areas & Qt::RightDockWidgetArea);
217 topAction->setEnabled(areas & Qt::TopDockWidgetArea);
218 bottomAction->setEnabled(areas & Qt::BottomDockWidgetArea);
219 }
220 }

223 {
224 if (!p)

231 allowRightAction->setEnabled(area != Qt::RightDockWidgetArea);
232 allowTopAction->setEnabled(area != Qt::TopDockWidgetArea);
233 allowBottomAction->setEnabled(area != Qt::BottomDockWidgetArea);
234 }
