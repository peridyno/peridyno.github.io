106 // find distance x0 is from segment x1-x2
107 static float point_segment_distance(const Vec3f &x0, const Vec3f &x1, const Vec3f &x2)
108 {
109 Vec3f dx(x2 - x1);
110 double m2 = dx.normSquared();
111 // find parameter value of closest point on segment
112 float s12 = (float)((x2 - x0).dot(dx) / m2);
113 if (s12 < 0) {
114 s12 = 0;
115 }
116 else if (s12 > 1) {
117 s12 = 1;
118 }
119 // and find the distance
120 return (x0 - (s12*x1 + (1 - s12)*x2)).norm();
121 }

123 // find distance x0 is from triangle x1-x2-x3
124 static float point_triangle_distance(const Vec3f &x0, const Vec3f &x1, const Vec3f &x2, const Vec3f &x3)
125 {
126 // first find barycentric coordinates of closest point on infinite plane
127 Vec3f x13(x1 - x3), x23(x2 - x3), x03(x0 - x3);
128 float m13 = x13.normSquared(), m23 = x23.normSquared(), d = x13.dot(x23);
129 float invdet = 1.f / std::max(m13*m23 - d * d, 1e-30f);
130 float a = x13.dot(x03), b = x23.dot(x03);
131 // the barycentric coordinates themselves
132 float w23 = invdet * (m23*a - d * b);
133 float w31 = invdet * (m13*b - d * a);
134 float w12 = 1 - w23 - w31;
135 if (w23 >= 0 && w31 >= 0 && w12 >= 0) { // if we're inside the triangle
136 return (x0 - (w23*x1 + w31 * x2 + w12 * x3)).norm();
137 }
138 else { // we have to clamp to one of the edges
139 if (w23 > 0) // this rules out edge 2-3 for us
140 return std::min(point_segment_distance(x0, x1, x2), point_segment_distance(x0, x1, x3));
141 else if (w31 > 0) // this rules out edge 1-3
142 return std::min(point_segment_distance(x0, x1, x2), point_segment_distance(x0, x2, x3));
143 else // w12 must be >0, ruling out edge 1-2
144 return std::min(point_segment_distance(x0, x1, x3), point_segment_distance(x0, x2, x3));
145 }

203 // calculate twice signed area of triangle (0,0)-(x1,y1)-(x2,y2)
204 // return an SOS-determined sign (-1, +1, or 0 only if it's a truly degenerate triangle)
205 static int orientation(double x1, double y1, double x2, double y2, double &twice_signed_area)
206 {
207 twice_signed_area = y1 * x2 - x1 * y2;
208 if (twice_signed_area > 0) return 1;
209 else if (twice_signed_area < 0) return -1;
210 else if (y2 > y1) return 1;
211 else if (y2 < y1) return -1;
212 else if (x1 > x2) return 1;
213 else if (x1 < x2) return -1;
214 else return 0; // only true when x1==x2 and y1==y2
215 }

217 // robust test of (x0,y0) in the triangle (x1,y1)-(x2,y2)-(x3,y3)
218 // if true is returned, the barycentric coordinates are set in a,b,c.
219 static bool point_in_triangle_2d(double x0, double y0,
220 double x1, double y1, double x2, double y2, double x3, double y3,
221 double& a, double& b, double& c)
222 {
223 x1 -= x0; x2 -= x0; x3 -= x0;
224 y1 -= y0; y2 -= y0; y3 -= y0;
225 int signa = orientation(x2, y2, x3, y3, a);
226 if (signa == 0) return false;
227 int signb = orientation(x3, y3, x1, y1, b);
228 if (signb != signa) return false;
229 int signc = orientation(x1, y1, x2, y2, c);
230 if (signc != signa) return false;
231 double sum = a + b + c;
232 assert(sum != 0); // if the SOS signs match and are nonkero, there's no way all of a, b, and c are zero.
233 a /= sum;
234 b /= sum;
235 c /= sum;
236 return true;
237 }
