285

292 if (inNode != nullptr && !visited[inNode->objectId()]) {
293 DFS(inNode, nodeQueue, visited);

295 }
296 }
297

355 {
356 if (!mQueueUpdateRequired)
357 return;
358
359 mNodeQueue.clear();
