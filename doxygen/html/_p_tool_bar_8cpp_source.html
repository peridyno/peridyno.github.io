136 static QPixmap genIcon(const QSize &iconSize, const QString &, const QColor &color)
137 {
138 int w = iconSize.width();
139 int h = iconSize.height();
140
141 QImage image(w, h, QImage::Format_ARGB32_Premultiplied);
142 image.fill(0);
143
144 QPainter p(&image);
145
146 extern void render_qt_text(QPainter *, int, int, const QColor &);
147 render_qt_text(&p, w, h, color);
148
149 return QPixmap::fromImage(image, Qt::DiffuseDither | Qt::DiffuseAlphaDither);
150 }

280 void PToolBar::updateMenu()
281 {
282 QMainWindow *mainWindow = qobject_cast<QMainWindow *>(parentWidget());
283 Q_ASSERT(mainWindow != 0);
284
285 const Qt::ToolBarArea area = mainWindow->toolBarArea(this);
286 const Qt::ToolBarAreas areas = allowedAreas();
287
288 movableAction->setChecked(isMovable());
289
290 allowLeftAction->setChecked(isAreaAllowed(Qt::LeftToolBarArea));
291 allowRightAction->setChecked(isAreaAllowed(Qt::RightToolBarArea));
292 allowTopAction->setChecked(isAreaAllowed(Qt::TopToolBarArea));
293 allowBottomAction->setChecked(isAreaAllowed(Qt::BottomToolBarArea));
294
295 if (allowedAreasActions->isEnabled()) {
296 allowLeftAction->setEnabled(area != Qt::LeftToolBarArea);
297 allowRightAction->setEnabled(area != Qt::RightToolBarArea);
298 allowTopAction->setEnabled(area != Qt::TopToolBarArea);
299 allowBottomAction->setEnabled(area != Qt::BottomToolBarArea);
300 }
301
302 leftAction->setChecked(area == Qt::LeftToolBarArea);
303 rightAction->setChecked(area == Qt::RightToolBarArea);
304 topAction->setChecked(area == Qt::TopToolBarArea);
305 bottomAction->setChecked(area == Qt::BottomToolBarArea);
306
307 if (areaActions->isEnabled()) {
308 leftAction->setEnabled(areas & Qt::LeftToolBarArea);
309 rightAction->setEnabled(areas & Qt::RightToolBarArea);
310 topAction->setEnabled(areas & Qt::TopToolBarArea);
311 bottomAction->setEnabled(areas & Qt::BottomToolBarArea);
312 }

315 void PToolBar::order()
316 {
317 QList<QAction *> ordered;
318 QList<QAction *> actions1 = actions();
319 foreach(QAction *action, findChildren<QAction *>()) {
320 if (!actions1.contains(action))
321 continue;
322 actions1.removeAll(action);
323 ordered.append(action);
324 }
325
326 clear();
327 addActions(ordered);
328
329 orderAction->setEnabled(false);
330 }

332 void PToolBar::randomize()
333 {
334 QList<QAction *> randomized;
335 QList<QAction *> actions = this->actions();
336 while (!actions.isEmpty()) {
337 QAction *action = actions.takeAt(rand() % actions.size());
338 randomized.append(action);
339 }
340 clear();
341 addActions(randomized);
342
343 orderAction->setEnabled(true);
344 }
