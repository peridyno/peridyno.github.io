403 void GlfwRenderWindow::mouseButtonCallback(GLFWwindow* window, int button, int action, int mods)
404 {
405 double xpos, ypos;
406 glfwGetCursorPos(window, &xpos, &ypos);
407
408 GlfwRenderWindow* activeWindow = (GlfwRenderWindow*)glfwGetWindowUserPointer(window);
409
410 // handle picking
411 if (activeWindow->getButtonType() == GLFW_MOUSE_BUTTON_LEFT &&
412 activeWindow->getButtonAction() == GLFW_PRESS &&
413 activeWindow->getButtonMode() == 0 &&
414 action == GLFW_RELEASE) {
415
416 // in picking
417 int x = fmin(xpos, activeWindow->getCursorPosX());
418 int y = fmax(ypos, activeWindow->getCursorPosY());
419 int w = fabs(xpos - activeWindow->getCursorPosX());
420 int h = fabs(ypos - activeWindow->getCursorPosY());
421 // flip y to texture space...
422 y = activeWindow->getHeight() - y - 1;
423
424 auto items = activeWindow->mRenderEngine->select(x, y, w, h);
425 // print selected result...
426 printf("Picking: (%d, %d) - (%d, %d), %d items...\n", x, y, w, h, items.size());
427
428 if (!items.empty()) {
429 // pick the last one?
430 auto node = items[0].node;
431 int instance = items[0].instance;
432
433 auto bbox = node->boundingBox();
434
435 printf(" Node: %s, bbox = (%.3f, %.3f, %.3f) - (%.3f, %.3f, %.3f)\n", node->getName().c_str(),
436 bbox.lower[0], bbox.lower[1], bbox.lower[2],
437 bbox.upper[0], bbox.upper[1], bbox.upper[2]);
438
439 // set selected node
440 activeWindow->currNode = node;
441 }
442 else {
443 activeWindow->currNode = 0;
444 }
445 }
446
447 auto camera = activeWindow->getCamera();
448
449 activeWindow->setButtonType(button);
450 activeWindow->setButtonAction(action);
451 activeWindow->setButtonMode(mods);
452
453 PMouseEvent mouseEvent;
454 mouseEvent.ray = camera->castRayInWorldSpace((float)xpos, (float)ypos);
455 mouseEvent.buttonType = (PButtonType)button;
456 mouseEvent.actionType = (PActionType)action;
457 mouseEvent.mods = (PModifierBits)activeWindow->getButtonMode();
458 mouseEvent.camera = camera;
459 mouseEvent.x = (float)xpos;
460 mouseEvent.y = (float)ypos;
461
462 auto activeScene = SceneGraphFactory::instance()->active();
463
464 activeScene->onMouseEvent(mouseEvent);
465
466 if (action == GLFW_PRESS)
467 {
468 // if(mOpenCameraRotate)
469 camera->registerPoint((float)xpos, (float)ypos);
470 activeWindow->setButtonState(GLFW_DOWN);
471
472 activeWindow->imWindow()->mousePressEvent(mouseEvent);
473 }
474 else
475 glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
476
477 if (action == GLFW_RELEASE)
478 {
479 activeWindow->setButtonState(GLFW_UP);
480
481 activeWindow->imWindow()->mouseReleaseEvent(mouseEvent);
482 }
483
484
485 // update cursor position record
486 if (action == GLFW_PRESS)
487 activeWindow->setCursorPos(xpos, ypos);
488 else
489 activeWindow->setCursorPos(-1, -1);
490 }
