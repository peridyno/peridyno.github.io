86 std::array<glm::vec4, 8> getFrustumCorners(const glm::mat4& proj)
87 {
88 const glm::vec4 p[8] = {
89 glm::vec4(-1.0f, -1.0f, -1.0f, 1.0f),
90 glm::vec4(-1.0f, -1.0f, 1.0f, 1.0f),
91
92 glm::vec4(-1.0f, 1.0f, -1.0f, 1.0f),
93 glm::vec4(-1.0f, 1.0f, 1.0f, 1.0f),
94
95 glm::vec4(1.0f, -1.0f, -1.0f, 1.0f),
96 glm::vec4(1.0f, -1.0f, 1.0f, 1.0f),
97
98 glm::vec4(1.0f, 1.0f, -1.0f, 1.0f),
99 glm::vec4(1.0f, 1.0f, 1.0f, 1.0f),
100 };
101
102 const glm::mat4 invProj = glm::inverse(proj);
103
104 std::array<glm::vec4, 8> corners;
105 for (int i = 0; i < 8; i++)
106 {
107 // camera space corners
108 corners[i] = invProj * p[i];
109 corners[i] /= corners[i].w;
110 }
111
112 return corners;
113 }

115 glm::mat4 getLightViewMatrix(glm::vec3 lightDir)
116 {
117 glm::vec3 lightUp = glm::vec3(0, 1, 0);
118 if (glm::length(glm::cross(lightUp, lightDir)) == 0.f)
119 {
120 lightUp = glm::vec3(0, 0, 1);
121 }
122 glm::mat4 lightView = glm::lookAt(glm::vec3(0), -lightDir, lightUp);
123 return lightView;
124 }

126 glm::mat4 getLightProjMatrix(glm::mat4 lightView,
127 Vec3f lowerBound,
128 Vec3f upperBound,
129 Camera* camera)
130 {
131 glm::vec4 p[8] = {
132 lightView * glm::vec4{lowerBound[0], lowerBound[1], lowerBound[2], 1},
133 lightView * glm::vec4{lowerBound[0], lowerBound[1], upperBound[2], 1},
134 lightView * glm::vec4{lowerBound[0], upperBound[1], lowerBound[2], 1},
135 lightView * glm::vec4{lowerBound[0], upperBound[1], upperBound[2], 1},
136 lightView * glm::vec4{upperBound[0], lowerBound[1], lowerBound[2], 1},
137 lightView * glm::vec4{upperBound[0], lowerBound[1], upperBound[2], 1},
138 lightView * glm::vec4{upperBound[0], upperBound[1], lowerBound[2], 1},
139 lightView * glm::vec4{upperBound[0], upperBound[1], upperBound[2], 1},
140 };
141
142 glm::vec4 bmin = p[0];
143 glm::vec4 bmax = p[0];
144 for (int i = 1; i < 8; i++)
145 {
146 bmin = glm::min(bmin, p[i]);
147 bmax = glm::max(bmax, p[i]);
148 }
149
150 // frustrum clamp
151 if (camera != 0)
152 {
153 std::array<glm::vec4, 8> corners = getFrustumCorners(camera->getProjMat());
154 glm::mat4 tm = lightView * glm::inverse(camera->getViewMat());
155
156 glm::vec4 fbmin = tm * corners[0];
157 glm::vec4 fbmax = tm * corners[0];
158 for (int i = 1; i < 8; i++)
159 {
160 glm::vec4 c = tm * corners[i];
161 fbmin = glm::min(fbmin, c);
162 fbmax = glm::max(fbmax, c);
163 }
164
165 bmin.x = glm::max(bmin.x, fbmin.x);
166 bmin.y = glm::max(bmin.y, fbmin.y);
167 bmax.x = glm::min(bmax.x, fbmax.x);
168 bmax.y = glm::min(bmax.y, fbmax.y);
169 }
170
171 float cx = (bmin.x + bmax.x) * 0.5;
172 float cy = (bmin.y + bmax.y) * 0.5;
173 float d = glm::max(bmax.y - bmin.y, bmax.x - bmin.x) * 0.5f;
174
175 glm::mat4 lightProj = glm::ortho(cx - d, cx + d, cy - d, cy + d, -bmax.z, -bmin.z);
176 return lightProj;
177 }
