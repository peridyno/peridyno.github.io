49 template <typename Real>
50 DYN_FUNC Quat<Real>::Quat(const Real yaw, const Real pitch, const Real roll)
51 {
52 Real cy = glm::cos(Real(yaw * 0.5));
53 Real sy = glm::sin(Real(yaw * 0.5));
54 Real cp = glm::cos(Real(pitch * 0.5));
55 Real sp = glm::sin(Real(pitch * 0.5));
56 Real cr = glm::cos(Real(roll * 0.5));
57 Real sr = glm::sin(Real(roll * 0.5));
58
59 w = cr * cp * cy + sr * sp * sy;
60 x = sr * cp * cy - cr * sp * sy;
61 y = cr * sp * cy + sr * cp * sy;
62 z = cr * cp * sy - sr * sp * cy;
63 }

172 DYN_FUNC Quat<Real>& Quat<Real>::normalize()
173 {
174 Real d = norm();
175 // Set the rotation along the x-axis
176 if (d < 0.00001) {
177 z = Real(1.0);
178 x = y = w = Real(0.0);
179 return *this;
180 }
181 d = Real(1) / d;
182 x *= d;
183 y *= d;
184 z *= d;
185 w *= d;
186 return *this;

195 template <typename Real>
196 DYN_FUNC void Quat<Real>::toEulerAngle(Real& yaw, Real& pitch, Real& roll) const
197 {
198 // roll (x-axis rotation)
199 Real sinr_cosp = 2 * (w * x + y * z);
200 Real cosr_cosp = 1 - 2 * (x * x + y * y);
201 roll = atan2(sinr_cosp, cosr_cosp);
202
203 // pitch (y-axis rotation)
204 Real sinp = 2 * (w * y - z * x);
205 if (glm::abs(sinp) >= 1)
206 pitch = sinp > 0 ? Real(M_PI / 2) : -Real(M_PI / 2); // use 90 degrees if out of range
207 else
208 pitch = glm::asin(sinp);
209
210 // yaw (z-axis rotation)
211 Real siny_cosp = 2 * (w * z + x * y);
212 Real cosy_cosp = 1 - 2 * (y * y + z * z);
213 yaw = atan2(siny_cosp, cosy_cosp);
214 }

241 template <typename Real>
242 DYN_FUNC Vector<Real, 3> Quat<Real>::rotate(const Vector<Real, 3>& v) const
243 {
244 // Extract the vector part of the quaternion
245 Vector<Real, 3> u(x, y, z);
246
247 // Extract the scalar part of the quaternion
248 Real s = w;
249
250 // Do the math
251 return 2.0f * u.dot(v) * u
252 + (s*s - u.dot(u)) * v
253 + 2.0f * s * u.cross(v);
254 }

300 template <typename Real>
301 DYN_FUNC Quat<Real>::Quat(const SquareMatrix<Real, 3>& matrix)
302 {
303 Real tr = matrix(0, 0) + matrix(1, 1) + matrix(2, 2);
304 if (tr > 0.0)
305 {
306 Real s = glm::sqrt(tr + Real(1.0));
307 w = s * Real(0.5);
308 if (s != 0.0)
309 s = Real(0.5) / s;
310 x = s * (matrix(1, 2) - matrix(2, 1));
311 y = s * (matrix(2, 0) - matrix(0, 2));
312 z = s * (matrix(0, 1) - matrix(1, 0));
313 }
314 else
315 {
316 int i = 0, j, k;
317 int next[3] = { 1, 2, 0 };
318 Real q[4];
319 if (matrix(1, 1) > matrix(0, 0)) i = 1;
320 if (matrix(2, 2) > matrix(i, i)) i = 2;
321 j = next[i];
322 k = next[j];
323 Real s = glm::sqrt(matrix(i, i) - matrix(j, j) - matrix(k, k) + Real(1.0));
324 q[i] = s * Real(0.5);
325 if (s != 0.0)
326 s = Real(0.5) / s;
327 q[3] = s * (matrix(j, k) - matrix(k, j));
328 q[j] = s * (matrix(i, j) - matrix(j, i));
329 q[k] = s * (matrix(i, k) - matrix(k, i));
330 x = q[0];
331 y = q[1];
332 z = q[2];
333 w = q[3];
334 }
335 }
