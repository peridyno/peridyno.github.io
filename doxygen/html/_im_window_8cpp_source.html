426void dyno::ImWindow::drawNodeManipulator(std::shared_ptr<Node> n, glm::mat4 view, glm::mat4 proj)
427{
428 // TODO: type conversion...
429 auto node = std::dynamic_pointer_cast<ParametricModel<DataType3f>>(n);
430
431 // TODO: parameterized operation
432 auto nodeOp = ImGuizmo::TRANSLATE;
433
434 switch (mEditMode) {
435 case 0:
436 nodeOp = ImGuizmo::TRANSLATE;
437 break;
438 case 1:
439 nodeOp = ImGuizmo::SCALE;
440 break;
441 case 2:
442 nodeOp = ImGuizmo::ROTATE;
443 break;
444 default:
445 break;
446 }
447
448 if (node != 0) {
449 float m[16];
450 // build transform matrix
451 {
452 auto t = node->varLocation()->getData();
453 auto s = node->varScale()->getData();
454 auto r = node->varRotation()->getData();
455 ImGuizmo::RecomposeMatrixFromComponents(t.getDataPtr(), r.getDataPtr(), s.getDataPtr(), m);
456 }
457
458 if (ImGuizmo::Manipulate(&view[0][0], &proj[0][0], nodeOp, ImGuizmo::WORLD, m, NULL, NULL, NULL, NULL))
459 {
460 float t[3], s[3], r[3];
461 ImGuizmo::DecomposeMatrixToComponents(m, t, r, s);
462 // apply
463 if (nodeOp == ImGuizmo::TRANSLATE)
464 node->varLocation()->setValue(Vec3f(t[0], t[1], t[2]));
465 if (nodeOp == ImGuizmo::SCALE)
466 node->varScale()->setValue(Vec3f(s[0], s[1], s[2]));
467 if (nodeOp == ImGuizmo::ROTATE)
468 node->varRotation()->setValue(Vec3f(r[0], r[1], r[2]));
469
470 // notify the update of node?
471 node->updateGraphicsContext();
472 }
473 }

476void dyno::ImWindow::drawViewManipulator(Camera* camera)
477{
478 glm::mat4 view = camera->getViewMat();
479 glm::mat4 view0 = view;
480 float dist = (camera->getEyePos() - camera->getTargetPos()).norm() * camera->unitScale();
481
482 ImGuizmo::ViewManipulate(&view[0][0], dist,
483 ImVec2(0, camera->viewportHeight() - 100),
484 ImVec2(100, 100),
485 0);
486
487 // if nothing changes
488 if (view == view0) return;
489
490 glm::mat4 invView = glm::inverse(view);
491 glm::vec4 eye = invView * glm::vec4(0, 0, 0, 1) / camera->unitScale();
492
493 // for trackball camera, also update up direction
494 TrackballCamera* cam = dynamic_cast<TrackballCamera*>(camera);
495 if (cam) {
496 glm::vec4 up = invView * glm::vec4(0, 1, 0, 0);
497 cam->mCameraUp = {up.x, up.y, up.z};
498
499 cam->setEyePos({ eye.x, eye.y, eye.z });
500 }
501 else
502 {
503 OrbitCamera* cam = dynamic_cast<OrbitCamera*>(camera);
504 // for orbit camera, simply set eye position...
505 cam->setEyePos({ eye.x, eye.y, eye.z });
506
507 // TODO: fix the problem when view along the rotation axis
508 }
509}
