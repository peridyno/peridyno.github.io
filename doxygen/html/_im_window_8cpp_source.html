358void ImWindow::drawSelectedRegion()
359{
360 if (mDrawingBox &&
361 mButtonType == BT_LEFT &&
362 mButtonAction == AT_PRESS &&
363 mButtonMode != MB_ALT &&
364 !ImGuizmo::IsUsing() &&
365 !ImGui::GetIO().WantCaptureMouse) {
366
367 ImVec2 pMin = { fminf(mRegX, mCurX), fminf(mRegY, mCurY) };
368 ImVec2 pMax = { fmaxf(mRegX, mCurX), fmaxf(mRegY, mCurY) };
369
370 // visible rectangle
371 if (pMin.x != pMax.x || pMin.y != pMax.y) {
372 // fill
373 ImGui::GetBackgroundDrawList()->AddRectFilled(pMin, pMax, ImColor{ 0.2f, 0.2f, 0.2f, 0.5f });
374 // border
375 ImGui::GetBackgroundDrawList()->AddRect(pMin, pMax, ImColor{ 0.8f, 0.8f, 0.8f, 0.8f }, 0, 0, 1.5f);
