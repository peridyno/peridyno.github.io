478void dyno::ImWindow::drawNodeManipulator(std::shared_ptr<Node> n, glm::mat4 view, glm::mat4 proj)
479{
480 // TODO: type conversion...
481 auto node = std::dynamic_pointer_cast<ParametricModel<DataType3f>>(n);
482
483 if (!node) return;
484
485 // get original transform
486 dyno::Vec3f t0 = node->varLocation()->getData();
487 dyno::Vec3f r0 = node->varRotation()->getData();
488 dyno::Vec3f s0 = node->varScale()->getData();
489
490 auto op = ImGuizmo::TRANSLATE;
491 auto mode = ImGuizmo::WORLD;
492
493 switch (mEditMode) {
494 case 0:
495 op = ImGuizmo::TRANSLATE;
496 break;
497 case 1:
498 op = ImGuizmo::SCALE;
499 mode = ImGuizmo::LOCAL;
500 break;
501 case 2:
502 op = ImGuizmo::ROTATE;
503 break;
504 default:
505 break;
506 }
507 if (node != 0) {
508 // build transform matrix
509 glm::mat4 m;
510 ImGuizmo::RecomposeMatrixFromComponents(t0.getDataPtr(), r0.getDataPtr(), s0.getDataPtr(), &m[0][0]);
511 if (ImGuizmo::Manipulate(&view[0][0], &proj[0][0], op, mode, &m[0][0], NULL, NULL, NULL, NULL))
512 {
513 dyno::Vec3f t1;
514 dyno::Vec3f r1;
515 dyno::Vec3f s1;
516
517 ImGuizmo::DecomposeMatrixToComponents(&m[0][0], t1.getDataPtr(), r1.getDataPtr(), s1.getDataPtr());
518
519 if (op == ImGuizmo::SCALE) {
520 // scale apply in local space
521 node->varScale()->setValue(s1);
522 }
523 if (op == ImGuizmo::TRANSLATE) {
524 node->varLocation()->setValue(t1);
525 }
526
527 if (op == ImGuizmo::ROTATE) {
528 node->varRotation()->setValue(r1);
529 }
530
531 node->updateGraphicsContext();
532 }
533 }

536void ImWindow::drawViewManipulator(Camera* camera)
537{
538 glm::mat4 view = camera->getViewMat();
539 glm::mat4 view0 = view;
540 float dist = (camera->getEyePos() - camera->getTargetPos()).norm() * camera->unitScale();
541
542 ImGuizmo::ViewManipulate(&view[0][0], dist,
543 ImVec2(0, camera->viewportHeight() - 100),
544 ImVec2(100, 100),
545 0);
546
547 // if nothing changes
548 if (view == view0) return;
549
550 glm::mat4 invView = glm::inverse(view);
551 glm::vec4 eye = invView * glm::vec4(0, 0, 0, 1) / camera->unitScale();
552
553 // for trackball camera, also update up direction
554 TrackballCamera* cam = dynamic_cast<TrackballCamera*>(camera);
555 if (cam) {
556 glm::vec4 up = invView * glm::vec4(0, 1, 0, 0);
557 cam->mCameraUp = {up.x, up.y, up.z};
558
559 cam->setEyePos({ eye.x, eye.y, eye.z });
560 }
561 else
562 {
563 OrbitCamera* cam = dynamic_cast<OrbitCamera*>(camera);
564 // for orbit camera, simply set eye position...
565 cam->setEyePos({ eye.x, eye.y, eye.z });
566
567 // TODO: fix the problem when view along the rotation axis
568 }
