92void Image::flipHorizontally()
93{
94 unsigned int pixel_size = pixelSize();
95 for(unsigned int i = 0; i < width_/2; ++i)
96 for(unsigned int j = 0; j < height_; ++j)
97 for(unsigned int k = 0; k < pixel_size; ++k)
98 {
99 unsigned char temp = raw_data_[(i+j*width_)*pixel_size+k];
100 raw_data_[(i+j*width_)*pixel_size+k] = raw_data_[((width_-1-i)+j*width_)*pixel_size+k];
101 raw_data_[((width_-1-i)+j*width_)*pixel_size+k] = temp;
102 }
103}

105void Image::flipVertically()
106{
107 unsigned int pixel_size = pixelSize();
108 for(unsigned int j = 0; j < height_/2; ++j)
109 for(unsigned int i = 0; i < pixel_size*width_; ++i)
110 {
111 unsigned char temp = raw_data_[i+j*width_*pixel_size];
112 raw_data_[i+j*width_*pixel_size] = raw_data_[i+ (height_-1-j)*width_*pixel_size];
113 raw_data_[i+ (height_-1-j)*width_*pixel_size] = temp;
114 }
115}
