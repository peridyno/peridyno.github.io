22 void updateViewMatrix()
23 {
24 glm::mat4 rotM = glm::mat4(1.0f);
25 glm::mat4 transM;
26
27 rotM = glm::rotate(rotM, glm::radians(rotation.x * (flipY ? -1.0f : 1.0f)), glm::vec3(1.0f, 0.0f, 0.0f));
28 rotM = glm::rotate(rotM, glm::radians(rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));
29 rotM = glm::rotate(rotM, glm::radians(rotation.z), glm::vec3(0.0f, 0.0f, 1.0f));
30
31 glm::vec3 translation = position;
32 if (flipY) {
33 translation.y *= -1.0f;
34 }
35 transM = glm::translate(glm::mat4(1.0f), translation);
36
37 if (type == CameraType::firstperson)
38 {
39 matrices.view = rotM * transM;
40 }
41 else
42 {
43 matrices.view = transM * rotM;
44 }
45
46 viewPos = glm::vec4(position, 0.0f) * glm::vec4(-1.0f, 1.0f, -1.0f, 1.0f);
47
48 updated = true;
49 };

150 void update(float deltaTime)
151 {
152 updated = false;
153 if (type == CameraType::firstperson)
154 {
155 if (moving())
156 {
157 glm::vec3 camFront;
158 camFront.x = -cos(glm::radians(rotation.x)) * sin(glm::radians(rotation.y));
159 camFront.y = sin(glm::radians(rotation.x));
160 camFront.z = cos(glm::radians(rotation.x)) * cos(glm::radians(rotation.y));
161 camFront = glm::normalize(camFront);
162
163 float moveSpeed = deltaTime * movementSpeed;
164
165 if (keys.up)
166 position += camFront * moveSpeed;
167 if (keys.down)
168 position -= camFront * moveSpeed;
169 if (keys.left)
170 position -= glm::normalize(glm::cross(camFront, glm::vec3(0.0f, 1.0f, 0.0f))) * moveSpeed;
171 if (keys.right)
172 position += glm::normalize(glm::cross(camFront, glm::vec3(0.0f, 1.0f, 0.0f))) * moveSpeed;
173
174 updateViewMatrix();
175 }
176 }

179 // Update camera passing separate axis data (gamepad)
180 // Returns true if view or position has been changed
181 bool updatePad(glm::vec2 axisLeft, glm::vec2 axisRight, float deltaTime)
182 {
183 bool retVal = false;
184
185 if (type == CameraType::firstperson)
186 {
187 // Use the common console thumbstick layout
188 // Left = view, right = move
189
190 const float deadZone = 0.0015f;
191 const float range = 1.0f - deadZone;
192
193 glm::vec3 camFront;
194 camFront.x = -cos(glm::radians(rotation.x)) * sin(glm::radians(rotation.y));
195 camFront.y = sin(glm::radians(rotation.x));
196 camFront.z = cos(glm::radians(rotation.x)) * cos(glm::radians(rotation.y));
197 camFront = glm::normalize(camFront);
198
199 float moveSpeed = deltaTime * movementSpeed * 2.0f;
200 float rotSpeed = deltaTime * rotationSpeed * 50.0f;
201
202 // Move
203 if (fabsf(axisLeft.y) > deadZone)
204 {
205 float pos = (fabsf(axisLeft.y) - deadZone) / range;
206 position -= camFront * pos * ((axisLeft.y < 0.0f) ? -1.0f : 1.0f) * moveSpeed;
207 retVal = true;
208 }
209 if (fabsf(axisLeft.x) > deadZone)
210 {
211 float pos = (fabsf(axisLeft.x) - deadZone) / range;
212 position += glm::normalize(glm::cross(camFront, glm::vec3(0.0f, 1.0f, 0.0f))) * pos * ((axisLeft.x < 0.0f) ? -1.0f : 1.0f) * moveSpeed;
213 retVal = true;
214 }
215
216 // Rotate
217 if (fabsf(axisRight.x) > deadZone)
218 {
219 float pos = (fabsf(axisRight.x) - deadZone) / range;
220 rotation.y += pos * ((axisRight.x < 0.0f) ? -1.0f : 1.0f) * rotSpeed;
221 retVal = true;
222 }
223 if (fabsf(axisRight.y) > deadZone)
224 {
225 float pos = (fabsf(axisRight.y) - deadZone) / range;
226 rotation.x -= pos * ((axisRight.y < 0.0f) ? -1.0f : 1.0f) * rotSpeed;
227 retVal = true;
228 }
229 }
230 else
231 {
232 // todo: move code from example base class for look-at
233 }
234
235 if (retVal)
236 {
237 updateViewMatrix();
238 }
239
240 return retVal;
241 }
