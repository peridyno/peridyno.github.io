254 template<typename Coord, typename Real, typename Constraint>
255 __global__ void checkOutError(
256 DArray<Coord> J,
257 DArray<Coord> mImpulse,
258 DArray<Constraint> constraints,
259 DArray<Real> eta,
260 DArray<Real> error
261 )
262 {
263 int tId = threadIdx.x + blockIdx.x * blockDim.x;
264 if (tId >= constraints.size())
265 return;
266
267 int idx1 = constraints[tId].bodyId1;
268 int idx2 = constraints[tId].bodyId2;
269
270 Real tmp = 0;
271 tmp += J[4 * tId].dot(mImpulse[idx1 * 2]) + J[4 * tId + 1].dot(mImpulse[idx1 * 2 + 1]);
272 if (idx2 != INVALID)
273 tmp += J[4 * tId + 2].dot(mImpulse[idx2 * 2]) + J[4 * tId + 3].dot(mImpulse[idx2 * 2 + 1]);
274
275 Real e = tmp - eta[tId];
276 error[tId] = e * e;
277 }
