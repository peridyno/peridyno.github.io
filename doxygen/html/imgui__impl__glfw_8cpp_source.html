110void ImGui_ImplGlfw_KeyCallback(GLFWwindow* window, int key, int scancode, int action, int mods)
111{
112 if (g_PrevUserCallbackKey != NULL)
113 g_PrevUserCallbackKey(window, key, scancode, action, mods);
114
115 ImGuiIO& io = ImGui::GetIO();
116 if (action == GLFW_PRESS)
117 io.KeysDown[key] = true;
118 if (action == GLFW_RELEASE)
119 io.KeysDown[key] = false;
120
121 // Modifiers are not reliable across systems
122 io.KeyCtrl = io.KeysDown[GLFW_KEY_LEFT_CONTROL] || io.KeysDown[GLFW_KEY_RIGHT_CONTROL];
123 io.KeyShift = io.KeysDown[GLFW_KEY_LEFT_SHIFT] || io.KeysDown[GLFW_KEY_RIGHT_SHIFT];
124 io.KeyAlt = io.KeysDown[GLFW_KEY_LEFT_ALT] || io.KeysDown[GLFW_KEY_RIGHT_ALT];
125#ifdef _WIN32
126 io.KeySuper = false;
127#else
128 io.KeySuper = io.KeysDown[GLFW_KEY_LEFT_SUPER] || io.KeysDown[GLFW_KEY_RIGHT_SUPER];
129#endif

239void ImGui_ImplGlfw_Shutdown()
240{
241 if (g_InstalledCallbacks)
242 {
243 glfwSetMouseButtonCallback(g_Window, g_PrevUserCallbackMousebutton);
244 glfwSetScrollCallback(g_Window, g_PrevUserCallbackScroll);
245 glfwSetKeyCallback(g_Window, g_PrevUserCallbackKey);
246 glfwSetCharCallback(g_Window, g_PrevUserCallbackChar);
247 g_InstalledCallbacks = false;
248 }
249
250 for (ImGuiMouseCursor cursor_n = 0; cursor_n < ImGuiMouseCursor_COUNT; cursor_n++)
251 {
252 glfwDestroyCursor(g_MouseCursors[cursor_n]);
253 g_MouseCursors[cursor_n] = NULL;
254 }
255 g_ClientApi = GlfwClientApi_Unknown;
256}

258static void ImGui_ImplGlfw_UpdateMousePosAndButtons()
259{
260 // Update buttons
261 ImGuiIO& io = ImGui::GetIO();
262 for (int i = 0; i < IM_ARRAYSIZE(io.MouseDown); i++)
263 {
264 // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
265 io.MouseDown[i] = g_MouseJustPressed[i] || glfwGetMouseButton(g_Window, i) != 0;
266 g_MouseJustPressed[i] = false;
267 }
268
269 // Update mouse position
270 const ImVec2 mouse_pos_backup = io.MousePos;
271 io.MousePos = ImVec2(-FLT_MAX, -FLT_MAX);
272#ifdef __EMSCRIPTEN__
273 const bool focused = true; // Emscripten
274#else
275 const bool focused = glfwGetWindowAttrib(g_Window, GLFW_FOCUSED) != 0;
276#endif
277 if (focused)
278 {
279 if (io.WantSetMousePos)
280 {
281 glfwSetCursorPos(g_Window, (double)mouse_pos_backup.x, (double)mouse_pos_backup.y);
282 }
283 else
284 {
285 double mouse_x, mouse_y;
286 glfwGetCursorPos(g_Window, &mouse_x, &mouse_y);
287 io.MousePos = ImVec2((float)mouse_x, (float)mouse_y);
288 }
289 }
290}

292static void ImGui_ImplGlfw_UpdateMouseCursor()
293{
294 ImGuiIO& io = ImGui::GetIO();
295 if ((io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange) || glfwGetInputMode(g_Window, GLFW_CURSOR) == GLFW_CURSOR_DISABLED)
296 return;
297
298 ImGuiMouseCursor imgui_cursor = ImGui::GetMouseCursor();
299 if (imgui_cursor == ImGuiMouseCursor_None || io.MouseDrawCursor)
300 {
301 // Hide OS mouse cursor if imgui is drawing it or if it wants no cursor
302 glfwSetInputMode(g_Window, GLFW_CURSOR, GLFW_CURSOR_HIDDEN);
303 }
304 else
305 {
306 // Show OS mouse cursor
307 // FIXME-PLATFORM: Unfocused windows seems to fail changing the mouse cursor with GLFW 3.2, but 3.3 works here.
308 glfwSetCursor(g_Window, g_MouseCursors[imgui_cursor] ? g_MouseCursors[imgui_cursor] : g_MouseCursors[ImGuiMouseCursor_Arrow]);
309 glfwSetInputMode(g_Window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
310 }
311}

313static void ImGui_ImplGlfw_UpdateGamepads()
314{
315 ImGuiIO& io = ImGui::GetIO();
316 memset(io.NavInputs, 0, sizeof(io.NavInputs));
317 if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0)
318 return;
319
320 // Update gamepad inputs
321 #define MAP_BUTTON(NAV_NO, BUTTON_NO) { if (buttons_count > BUTTON_NO && buttons[BUTTON_NO] == GLFW_PRESS) io.NavInputs[NAV_NO] = 1.0f; }
322 #define MAP_ANALOG(NAV_NO, AXIS_NO, V0, V1) { float v = (axes_count > AXIS_NO) ? axes[AXIS_NO] : V0; v = (v - V0) / (V1 - V0); if (v > 1.0f) v = 1.0f; if (io.NavInputs[NAV_NO] < v) io.NavInputs[NAV_NO] = v; }
323 int axes_count = 0, buttons_count = 0;
324 const float* axes = glfwGetJoystickAxes(GLFW_JOYSTICK_1, &axes_count);
325 const unsigned char* buttons = glfwGetJoystickButtons(GLFW_JOYSTICK_1, &buttons_count);
326 MAP_BUTTON(ImGuiNavInput_Activate, 0); // Cross / A
327 MAP_BUTTON(ImGuiNavInput_Cancel, 1); // Circle / B
328 MAP_BUTTON(ImGuiNavInput_Menu, 2); // Square / X
329 MAP_BUTTON(ImGuiNavInput_Input, 3); // Triangle / Y
330 MAP_BUTTON(ImGuiNavInput_DpadLeft, 13); // D-Pad Left
331 MAP_BUTTON(ImGuiNavInput_DpadRight, 11); // D-Pad Right
332 MAP_BUTTON(ImGuiNavInput_DpadUp, 10); // D-Pad Up
333 MAP_BUTTON(ImGuiNavInput_DpadDown, 12); // D-Pad Down
334 MAP_BUTTON(ImGuiNavInput_FocusPrev, 4); // L1 / LB
335 MAP_BUTTON(ImGuiNavInput_FocusNext, 5); // R1 / RB
336 MAP_BUTTON(ImGuiNavInput_TweakSlow, 4); // L1 / LB
337 MAP_BUTTON(ImGuiNavInput_TweakFast, 5); // R1 / RB
338 MAP_ANALOG(ImGuiNavInput_LStickLeft, 0, -0.3f, -0.9f);
339 MAP_ANALOG(ImGuiNavInput_LStickRight,0, +0.3f, +0.9f);
340 MAP_ANALOG(ImGuiNavInput_LStickUp, 1, +0.3f, +0.9f);
341 MAP_ANALOG(ImGuiNavInput_LStickDown, 1, -0.3f, -0.9f);
342 #undef MAP_BUTTON
343 #undef MAP_ANALOG
344 if (axes_count > 0 && buttons_count > 0)
345 io.BackendFlags |= ImGuiBackendFlags_HasGamepad;
346 else
347 io.BackendFlags &= ~ImGuiBackendFlags_HasGamepad;
348}

350void ImGui_ImplGlfw_NewFrame()
351{
352 ImGuiIO& io = ImGui::GetIO();
353 IM_ASSERT(io.Fonts->IsBuilt() && "Font atlas not built! It is generally built by the renderer backend. Missing call to renderer _NewFrame() function? e.g. ImGui_ImplOpenGL3_NewFrame().");
354
355 // Setup display size (every frame to accommodate for window resizing)
356 int w, h;
357 int display_w, display_h;
358 glfwGetWindowSize(g_Window, &w, &h);
359 glfwGetFramebufferSize(g_Window, &display_w, &display_h);
360 io.DisplaySize = ImVec2((float)w, (float)h);
361 if (w > 0 && h > 0)
362 io.DisplayFramebufferScale = ImVec2((float)display_w / w, (float)display_h / h);
363
364 // Setup time step
365 double current_time = glfwGetTime();
366 io.DeltaTime = g_Time > 0.0 ? (float)(current_time - g_Time) : (float)(1.0f / 60.0f);
367 g_Time = current_time;
368
369 ImGui_ImplGlfw_UpdateMousePosAndButtons();
370 ImGui_ImplGlfw_UpdateMouseCursor();
371
372 // Update game controllers (if enabled and available)
373 ImGui_ImplGlfw_UpdateGamepads();
374}
