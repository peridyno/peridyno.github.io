20 VKAPI_ATTR VkBool32 VKAPI_CALL debugUtilsMessengerCallback(
21 VkDebugUtilsMessageSeverityFlagBitsEXT messageSeverity,
22 VkDebugUtilsMessageTypeFlagsEXT messageType,
23 const VkDebugUtilsMessengerCallbackDataEXT* pCallbackData,
24 void* pUserData)
25 {
26 // Select prefix depending on flags passed to the callback
27 std::string prefix("");
28
29 if (messageSeverity & VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT) {
30 prefix = "VERBOSE: ";
31 }
32 else if (messageSeverity & VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT) {
33 prefix = "INFO: ";
34 }
35 else if (messageSeverity & VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT) {
36 prefix = "WARNING: ";
37 }
38 else if (messageSeverity & VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT) {
39 prefix = "ERROR: ";
40 }
41
42
43 // Display message to default output (console/logcat)
44 std::stringstream debugMessage;
45 debugMessage << prefix << "[" << pCallbackData->messageIdNumber << "][" << pCallbackData->pMessageIdName << "] : " << pCallbackData->pMessage;
46
47#if defined(__ANDROID__)
48 if (messageSeverity >= VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT) {
49 LOGE("%s", debugMessage.str().c_str());
50 } else {
51 LOGD("%s", debugMessage.str().c_str());
52 }
53#else
54 if (messageSeverity >= VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT) {
55 std::cerr << debugMessage.str() << "\n";
56 } else {
57 std::cout << debugMessage.str() << "\n";
58 }
59 fflush(stdout);
60#endif
61
62
63 // The return value of this callback controls whether the Vulkan call that caused the validation message will be aborted or not
64 // We return VK_FALSE as we DON'T want Vulkan calls that cause a validation message to abort
65 // If you instead want to have calls abort, pass in VK_TRUE and the function will return VK_ERROR_VALIDATION_FAILED_EXT
66 return VK_FALSE;
67 }

69 void setupDebugging(VkInstance instance, VkDebugReportFlagsEXT flags, VkDebugReportCallbackEXT callBack)
70 {
71
72 vkCreateDebugUtilsMessengerEXT = reinterpret_cast<PFN_vkCreateDebugUtilsMessengerEXT>(vkGetInstanceProcAddr(instance, "vkCreateDebugUtilsMessengerEXT"));
73 vkDestroyDebugUtilsMessengerEXT = reinterpret_cast<PFN_vkDestroyDebugUtilsMessengerEXT>(vkGetInstanceProcAddr(instance, "vkDestroyDebugUtilsMessengerEXT"));
74
75 VkDebugUtilsMessengerCreateInfoEXT debugUtilsMessengerCI{};
76 debugUtilsMessengerCI.sType = VK_STRUCTURE_TYPE_DEBUG_UTILS_MESSENGER_CREATE_INFO_EXT;
77 debugUtilsMessengerCI.messageSeverity = VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT | VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT | VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT;
78 debugUtilsMessengerCI.messageType = VK_DEBUG_UTILS_MESSAGE_TYPE_GENERAL_BIT_EXT | VK_DEBUG_UTILS_MESSAGE_TYPE_VALIDATION_BIT_EXT;
79 debugUtilsMessengerCI.pfnUserCallback = debugUtilsMessengerCallback;
80 VkResult result = vkCreateDebugUtilsMessengerEXT(instance, &debugUtilsMessengerCI, nullptr, &debugUtilsMessenger);
81 assert(result == VK_SUCCESS);
82 }

87 {
88 vkDestroyDebugUtilsMessengerEXT(instance, debugUtilsMessenger, nullptr);
