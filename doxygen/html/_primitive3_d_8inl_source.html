101 DYN_FUNC TPoint3D<Real> TPoint3D<Real>::project(const TTriangle3D<Real>& triangle) const
102 {
103 Coord3D dir = triangle.v[0] - origin;
104 Coord3D e0 = triangle.v[1] - triangle.v[0];
105 Coord3D e1 = triangle.v[2] - triangle.v[0];
106 Real a = e0.dot(e0);
107 Real b = e0.dot(e1);
108 Real c = e1.dot(e1);
109 Real d = e0.dot(dir);
110 Real e = e1.dot(dir);
111 Real f = dir.dot(dir);
112
113 Real det = a * c - b * b;
114 Real s = b * e - c * d;
115 Real t = b * d - a * e;
116
117 Real maxL = triangle.maximumEdgeLength();
118 //handle degenerate triangles
119 if (det < REAL_EPSILON * maxL * maxL)
120 {
121 Real g = (triangle.v[2] - triangle.v[1]).normSquared();
122
123 Real l_max = a;
124 Coord3D p0 = triangle.v[0];
125 Coord3D p1 = triangle.v[1];
126
127 if (c > l_max)
128 {
129 p0 = triangle.v[0];
130 p1 = triangle.v[2];
131
132 l_max = c;
133 }
134
135 if (g > l_max)
136 {
137 p0 = triangle.v[1];
138 p1 = triangle.v[2];
139 }
140
141 return project(TSegment3D<Real>(p0, p1));
142 }
143
144 if (s + t <= det)
145 {
146 if (s < 0)
147 {
148 if (t < 0)
149 {
150 //region 4
151 s = 0;
152 t = 0;
153 }
154 else
155 {
156 // region 3
157 // F(t) = Q(0, t) = ct^2 + 2et + f
158 // F'(t)/2 = ct + e
159 // F'(t) = 0 when t = -e/c
160
161 s = 0;
162 t = (e >= 0 ? 0 : (-e >= c ? 1 : -e / c));
163 }
164 }
165 else
166 {
167 if (t < 0)
168 {
169 //region 5
170 // F(s) = Q(s, 0) = as^2 + 2ds + f
171 // F'(s)/2 = as + d
172 // F'(s) = 0 when t = -d/a
173
174 s = (d >= 0 ? 0 : (-d >= a ? 1 : -d / a));
175 t = 0;
176 }
177 else
178 {
179 //region 0
180 Real invDet = 1 / det;
181 s *= invDet;
182 t *= invDet;
183 }
184 }
185 }
186 else
187 {
188 if (s < 0)
189 {
190 //region 2
191 s = 0;
192 t = 1;
193 }
194 else if (t < 0)
195 {
196 //region 6
197 s = 1;
198 t = 0;
199 }
200 else
201 {
202 //region 1
203 // F(s) = Q(s, 1 - s) = (a - 2b + c)s^2 + 2(b - c + d - e)s + (c + 2e + f)
204 // F'(s)/2 = (a - 2b + c)s + (b - c + d - e)
205 // F'(s} = 0 when s = (c + e - b - d)/(a - 2b + c)
206 // a - 2b + c = |e0 - e1|^2 > 0,
207 // so only the sign of c + e - b - d need be considered
208
209 Real numer = c + e - b - d;
210 if (numer <= 0) {
211 s = 0;
212 }
213 else {
214 Real denom = a - 2 * b + c; // positive quantity
215 s = (numer >= denom ? 1 : numer / denom);
216 }
217 t = 1 - s;
218 }
219 }
220 return TPoint3D<Real>(triangle.v[0] + s * e0 + t * e1);
221 }

980 template<typename Real>
981 DYN_FUNC TSegment3D<Real> TLine3D<Real>::proximity(const TTriangle3D<Real>& triangle) const
982 {
983 Coord3D e0 = triangle.v[1] - triangle.v[0];
984 Coord3D e1 = triangle.v[2] - triangle.v[0];
985 Coord3D normal = e0.cross(e1);
986 Real NdD = normal.dot(direction);
987 if (abs(NdD) > REAL_EPSILON)
988 {
989 // The line and triangle are not parallel, so the line
990 // intersects/ the plane of the triangle.
991 Coord3D diff = origin - triangle.v[0];
992
993 //compute a right - handed orthonormal basis
994 Coord3D W = direction;
995 W.normalize();
996 Coord3D U = W[0] > W[1] ? Coord3D(-W[2], (Real)0, W[0]) : Coord3D((Real)0, W[2], -W[1]);
997 Coord3D V = W.cross(U);
998
999 Real UdE0 = U.dot(e0);
1000 Real UdE1 = U.dot(e1);
1001 Real UdDiff = U.dot(diff);
1002 Real VdE0 = V.dot(e0);
1003 Real VdE1 = V.dot(e1);
1004 Real VdDiff = V.dot(diff);
1005 Real invDet = ((Real)1) / (UdE0 * VdE1 - UdE1 * VdE0);
1006
1007 // Barycentric coordinates for the point of intersection.
1008 Real u = (VdE1 * UdDiff - UdE1 * VdDiff) * invDet;
1009 Real v = (UdE0 * VdDiff - VdE0 * UdDiff) * invDet;
1010 Real b0 = (Real)1 - u - v;
1011
1012 if (b0 >= (Real)0 && u >= (Real)0 && v >= (Real)0)
1013 {
1014 // Line parameter for the point of intersection.
1015 Real DdE0 = direction.dot(e0);
1016 Real DdE1 = direction.dot(e1);
1017 Real DdDiff = direction.dot(diff);
1018 Real t = u * DdE0 + v * DdE1 - DdDiff;
1019
1020 return TSegment3D<Real>(origin + t * direction, triangle.v[0] + u * e0 + v * e1);
1021 }
1022 }
1023
1024 if (direction.normSquared() < EPSILON)
1025 {
1026 TPoint3D<Real> p(origin);
1027 return p.project(triangle) - p;
1028 }
1029 else
1030 {
1031 // Treat line and triangle as parallel. Compute
1032 // closest points by computing distance from
1033 // line to each triangle edge and taking minimum.
1034 TSegment3D<Real> e0(triangle.v[0], triangle.v[1]);
1035 TSegment3D<Real> minPQ = proximity(e0);
1036 Real minDS = minPQ.lengthSquared();
1037
1038 TSegment3D<Real> e1(triangle.v[0], triangle.v[2]);
1039 TSegment3D<Real> pq1 = proximity(e1);
1040 if (pq1.lengthSquared() < minDS)
1041 {
1042 minPQ = pq1;
1043 minDS = pq1.lengthSquared();
1044 }
1045
1046 TSegment3D<Real> e2(triangle.v[1], triangle.v[2]);
1047 TSegment3D<Real> pq2 = proximity(e2);
1048 if (pq2.lengthSquared() < minDS)
1049 {
1050 minPQ = pq2;
1051 }
1052
1053 return minPQ;
1054 }
1055 }

1329 if (tmp <= ((Real)2) * lenSqr * boxExtent[i2])
1330 {
1331 t = tmp / lenSqr;
1332 lenSqr += dir[i2] * dir[i2];
1333 tmp = PpE[i2] - t;
1334 delta = dir[i0] * PmE[i0] + dir[i1] * PpE[i1] + dir[i2] * tmp;
1335 param = -delta / lenSqr;
1336
1337 final_t = param;
1338 pnt[i0] = boxExtent[i0];
1339 pnt[i1] = -boxExtent[i1];
1340 pnt[i2] = t - boxExtent[i2];
1341 }
1342 else
1343 {
1344 lenSqr += dir[i2] * dir[i2];
1345 delta = dir[i0] * PmE[i0] + dir[i1] * PpE[i1]
1346 + dir[i2] * PmE[i2];
1347 param = -delta / lenSqr;
1348
1349 final_t = param;
1350 pnt[i0] = boxExtent[i0];
1351 pnt[i1] = -boxExtent[i1];
1352 pnt[i2] = boxExtent[i2];
1353 }
1354 return;
1355 }
1356
1357 // (v[i1],v[i2])-corner is closest
1358 lenSqr += dir[i2] * dir[i2];
1359 delta = dir[i0] * PmE[i0] + dir[i1] * PpE[i1] + dir[i2] * PpE[i2];
1360 param = -delta / lenSqr;
1361
1362 final_t = param;
1363 pnt[i0] = boxExtent[i0];
1364 pnt[i1] = -boxExtent[i1];
1365 pnt[i2] = -boxExtent[i2];
1366 }
1367 }
1368 };
1369
1370
1371 // Apply reflections so that direction vector has nonnegative
1372 // components.
1373 bool reflect[3];
1374 for (int i = 0; i < 3; ++i)
1375 {
1376 if (lineDir[i] < (Real)0)
1377 {
1378 point[i] = -point[i];
1379 lineDir[i] = -lineDir[i];
1380 reflect[i] = true;
1381 }
1382 else
1383 {
1384 reflect[i] = false;
1385 }
1386 }
1387
1388 if (lineDir[0] > REAL_EPSILON)
1389 {
1390 if (lineDir[1] > REAL_EPSILON)
1391 {
1392 if (lineDir[2] > REAL_EPSILON) // (+,+,+)
1393 {
1394 //CaseNoZeros(point, dir, boxExtent, t);
1395 Coord3D PmE = point - boxExtent;
1396 Real prodDxPy = lineDir[0] * PmE[1];
1397 Real prodDyPx = lineDir[1] * PmE[0];
1398 Real prodDzPx, prodDxPz, prodDzPy, prodDyPz;
1399
1400 if (prodDyPx >= prodDxPy)
1401 {
1402 prodDzPx = lineDir[2] * PmE[0];
1403 prodDxPz = lineDir[0] * PmE[2];
1404 if (prodDzPx >= prodDxPz)
1405 {
1406 // line intersects x = e0
1407 face(point, t, lineDir, PmE, boxExtent, 0, 1, 2);
1408 }
1409 else
1410 {
1411 // line intersects z = e2
1412 face(point, t, lineDir, PmE, boxExtent, 2, 0, 1);
1413 }
1414 }
1415 else
1416 {
1417 prodDzPy = lineDir[2] * PmE[1];
1418 prodDyPz = lineDir[1] * PmE[2];
1419 if (prodDzPy >= prodDyPz)
1420 {
1421 // line intersects y = e1
1422 face(point, t, lineDir, PmE, boxExtent, 1, 2, 0);
1423 }
1424 else
1425 {
1426 // line intersects z = e2
1427 face(point, t, lineDir, PmE, boxExtent, 2, 0, 1);
1428 }
1429 }
1430 }
1431 else // (+,+,0)
1432 {
1433 //Case0(0, 1, 2, point, dir, boxExtent, t);
1434 case0(point, t, lineDir, boxExtent, 0, 1, 2);
1435 }
1436 }
1437 else
1438 {
1439 if (lineDir[2] > REAL_EPSILON) // (+,0,+)
1440 {
1441 //Case0(0, 2, 1, point, dir, boxExtent, t);
1442 case0(point, t, lineDir, boxExtent, 0, 2, 1);
1443 }
1444 else // (+,0,0)
1445 {
1446 //Case00(0, 1, 2, point, dir, boxExtent, t);
1447 case00(point, t, lineDir, boxExtent, 0);
1448 }
1449 }
1450 }
1451 else
1452 {
1453 if (lineDir[1] > REAL_EPSILON)
1454 {
1455 if (lineDir[2] > REAL_EPSILON) // (0,+,+)
1456 {
1457 //Case0(1, 2, 0, point, dir, boxExtent, t);
1458 case0(point, t, lineDir, boxExtent, 1, 2, 0);
1459 }
1460 else // (0,+,0)
1461 {
1462 //Case00(1, 0, 2, point, dir, boxExtent, t);
1463 case00(point, t, lineDir, boxExtent, 1);
1464 }
1465 }
1466 else
1467 {
1468 if (lineDir[2] > REAL_EPSILON) // (0,0,+)
1469 {
1470 //Case00(2, 0, 1, point, dir, boxExtent, t);
1471 case00(point, t, lineDir, boxExtent, 2);
1472 }
1473 else // (0,0,0)
1474 {
1475 point = clamp(point, -boxExtent, boxExtent);
1476 //Case000(point, boxExtent);
1477 }
1478 }
1479 }
1480
1481 // Undo the reflections applied previously.
1482 for (int i = 0; i < 3; ++i)
1483 {
1484 if (reflect[i])
1485 {
1486 point[i] = -point[i];
1487 }
1488 }
1489
1490 return TSegment3D<Real>(origin + t * direction, boxCenter + point);
1491 }

1840 {

1842 Real a = direction.normSquared();
1843 Real b = direction.dot(ray.direction);
1844 Real c = ray.direction.normSquared();
1845 Real d = u.dot(direction);
1846 Real e = u.dot(ray.direction);

1848
1849 if (det < REAL_EPSILON)
1850 {
1851 TPoint3D<Real> p0(origin);
1852 TPoint3D<Real> p1(ray.origin);
1853
1854 TPoint3D<Real> q0 = p0.project(ray);
1855 TPoint3D<Real> q1 = p1.project(*this);
1856
1857 return (q0 - p0).lengthSquared() < (q1 - p1).lengthSquared() ? q0 - p0 : p1 - q1;
1858 }
1859
1860 Real sNum = b * e - c * d;
1861 Real tNum = a * e - b * d;
1862
1863 Real sDenom = det;
1864 Real tDenom = det;
1865
1866 if (sNum < 0) {
1867 sNum = 0;
1868 tNum = e;
1869 tDenom = c;
1870 }
1871 // Check t
1872 if (tNum < 0) {
1873 tNum = 0;
1874 if (-d < 0) {
1875 sNum = 0;
1876 }
1877 else {
1878 sNum = -d;
1879 sDenom = a;
1880 }
1881 }
1882 // Parameters of nearest points on restricted domain
1883 Real s = sNum / sDenom;
1884 Real t = tNum / tDenom;
1885

1891 {

1894 Real b = direction.dot(segment.direction());
1895 Real c = segment.lengthSquared();
1896 Real d = u.dot(direction);
1897 Real e = u.dot(segment.direction());
1898 Real det = a * c - b * b;
1899
1900 if (det < REAL_EPSILON)
1901 {
1902 if (a < REAL_EPSILON_SQUARED)
1903 {
1904 TPoint3D<Real> p0(origin);
1905 return p0.project(segment) - p0;
1906 }
1907 else
1908 {

1910 TPoint3D<Real> p2(segment.endPoint());
1911

1913 TPoint3D<Real> q2 = p2.project(*this);
1914
1915 return (p1 - q1).lengthSquared() < (p2 - q2).lengthSquared() ? (p1 - q1) : (p2 - q2);
1916 }
1917 }
1918
1919 Real sNum = b * e - c * d;
1920 Real tNum = a * e - b * d;
1921
1922 Real sDenom = det;
1923 Real tDenom = det;
1924
1925 if (sNum < 0) {
1926 sNum = 0;
1927 tNum = e;
1928 tDenom = c;
1929 }
1930

1932 if (tNum < 0) {
1933 tNum = 0;
1934 if (-d < 0) {
1935 sNum = 0;
1936 }
1937 else {
1938 sNum = -d;
1939 sDenom = a;
1940 }
1941 }
1942 else if (tNum > tDenom) {
1943 tNum = tDenom;

1945 sNum = 0;
1946 }
1947 else {
1948 sNum = -d + b;
1949 sDenom = a;
1950 }
1951 }

1953 Real s = sNum / sDenom;
1954 Real t = tNum / tDenom;
1955
