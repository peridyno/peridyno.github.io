245 void PDockWidget::updateContextMenu()
246 {
247 const Qt::DockWidgetArea area = mainWindow->dockWidgetArea(this);
248 const Qt::DockWidgetAreas areas = allowedAreas();
249
250 closableAction->setChecked(features() & QDockWidget::DockWidgetClosable);
251 if (windowType() == Qt::Drawer) {
252 floatableAction->setEnabled(false);
253 floatingAction->setEnabled(false);
254 movableAction->setEnabled(false);
255 verticalTitleBarAction->setChecked(false);
256 }
257 else {
258 floatableAction->setChecked(features() & QDockWidget::DockWidgetFloatable);
259 floatingAction->setChecked(isWindow());
260 // done after floating, to get 'floatable' correctly initialized
261 movableAction->setChecked(features() & QDockWidget::DockWidgetMovable);
262 verticalTitleBarAction
263 ->setChecked(features() & QDockWidget::DockWidgetVerticalTitleBar);
264 }
265
266 allowLeftAction->setChecked(isAreaAllowed(Qt::LeftDockWidgetArea));
267 allowRightAction->setChecked(isAreaAllowed(Qt::RightDockWidgetArea));
268 allowTopAction->setChecked(isAreaAllowed(Qt::TopDockWidgetArea));
269 allowBottomAction->setChecked(isAreaAllowed(Qt::BottomDockWidgetArea));
270
271 if (allowedAreasActions->isEnabled()) {
272 allowLeftAction->setEnabled(area != Qt::LeftDockWidgetArea);
273 allowRightAction->setEnabled(area != Qt::RightDockWidgetArea);
274 allowTopAction->setEnabled(area != Qt::TopDockWidgetArea);
275 allowBottomAction->setEnabled(area != Qt::BottomDockWidgetArea);
276 }
277
278 {
279 const QSignalBlocker blocker(leftAction);
280 leftAction->setChecked(area == Qt::LeftDockWidgetArea);
281 }
282 {
283 const QSignalBlocker blocker(rightAction);
284 rightAction->setChecked(area == Qt::RightDockWidgetArea);
285 }
286 {
287 const QSignalBlocker blocker(topAction);
288 topAction->setChecked(area == Qt::TopDockWidgetArea);
289 }
290 {
291 const QSignalBlocker blocker(bottomAction);
292 bottomAction->setChecked(area == Qt::BottomDockWidgetArea);
293 }
294
295 if (areaActions->isEnabled()) {
296 leftAction->setEnabled(areas & Qt::LeftDockWidgetArea);
297 rightAction->setEnabled(areas & Qt::RightDockWidgetArea);
298 topAction->setEnabled(areas & Qt::TopDockWidgetArea);
299 bottomAction->setEnabled(areas & Qt::BottomDockWidgetArea);
300 }
301
302 tabMenu->clear();
303 splitHMenu->clear();
304 splitVMenu->clear();
305 QList<PDockWidget*> dock_list = mainWindow->findChildren<PDockWidget*>();
306 foreach(PDockWidget *dock, dock_list) {
307 tabMenu->addAction(dock->objectName());
308 splitHMenu->addAction(dock->objectName());
309 splitVMenu->addAction(dock->objectName());
310 }
311 }

314 {
315 foreach(PDockWidget *dock, mainWindow->findChildren<PDockWidget*>()) {

322 void PDockWidget::splitInto(QAction *action)
323 {
324 PDockWidget *target = findByName(mainWindow, action->text());
325 if (!target)
326 return;
327

474 void BlueTitleBar::paintEvent(QPaintEvent*)
475 {
476 QPainter painter(this);
477 QRect rect = this->rect();
478
479 QDockWidget *dw = qobject_cast<QDockWidget*>(parentWidget());
480 Q_ASSERT(dw != 0);
481
482 if (dw->features() & QDockWidget::DockWidgetVerticalTitleBar) {
483 QSize s = rect.size();
484 s.transpose();
485 rect.setSize(s);
486
487 painter.translate(rect.left(), rect.top() + rect.width());
488 painter.rotate(-90);
489 painter.translate(-rect.left(), -rect.top());
490 }
491
492 painter.drawPixmap(rect.topLeft(), leftPm);
493 painter.drawPixmap(rect.topRight() - QPoint(rightPm.width() - 1, 0), rightPm);
494 QBrush brush(centerPm);
495 painter.fillRect(rect.left() + leftPm.width(), rect.top(),
496 rect.width() - leftPm.width() - rightPm.width(),
497 centerPm.height(), centerPm);
498 }

500 void BlueTitleBar::mouseReleaseEvent(QMouseEvent *event)
501 {
502 QPoint pos = event->pos();
503
504 QRect rect = this->rect();
505
506 QDockWidget *dw = qobject_cast<QDockWidget*>(parentWidget());
507 Q_ASSERT(dw != 0);
508
509 if (dw->features() & QDockWidget::DockWidgetVerticalTitleBar) {
510 QPoint p = pos;
511 pos.setX(rect.left() + rect.bottom() - p.y());
512 pos.setY(rect.top() + p.x() - rect.left());
513
514 QSize s = rect.size();
515 s.transpose();
516 rect.setSize(s);
517 }
518
519 const int buttonRight = 7;
520 const int buttonWidth = 20;
521 int right = rect.right() - pos.x();
522 int button = (right - buttonRight) / buttonWidth;
523 switch (button) {
524 case 0:
525 event->accept();
526 dw->close();
527 break;
528 case 1:
529 event->accept();
530 dw->setFloating(!dw->isFloating());
531 break;
532 case 2: {
533 event->accept();
534 QDockWidget::DockWidgetFeatures features = dw->features();
535 if (features & QDockWidget::DockWidgetVerticalTitleBar)
536 features &= ~QDockWidget::DockWidgetVerticalTitleBar;
537 else
538 features |= QDockWidget::DockWidgetVerticalTitleBar;
539 dw->setFeatures(features);
540 break;
541 }
542 default:
543 event->ignore();
544 break;
545 }
546 }

548 void BlueTitleBar::updateMask()
549 {
550 QDockWidget *dw = qobject_cast<QDockWidget*>(parent());
551 Q_ASSERT(dw != 0);
552
553 QRect rect = dw->rect();
554 QPixmap bitmap(dw->size());
555
556 {
557 QPainter painter(&bitmap);
558
559 // initialize to transparent
560 painter.fillRect(rect, Qt::color0);
561
562 QRect contents = rect;
563 contents.setTopLeft(geometry().bottomLeft());
564 contents.setRight(geometry().right());
565 contents.setBottom(contents.bottom() - y());
566 painter.fillRect(contents, Qt::color1);
567
568 // let's paint the titlebar
569 QRect titleRect = this->geometry();
570
571 if (dw->features() & QDockWidget::DockWidgetVerticalTitleBar) {
572 QSize s = rect.size();
573 s.transpose();
574 rect.setSize(s);
575
576 QSize s2 = size();
577 s2.transpose();
578 titleRect.setSize(s2);
579
580 painter.translate(rect.left(), rect.top() + rect.width());
581 painter.rotate(-90);
582 painter.translate(-rect.left(), -rect.top());
583 }
584
585 contents.setTopLeft(titleRect.bottomLeft());
586 contents.setRight(titleRect.right());
587 contents.setBottom(rect.bottom() - y());
588
589 QRect rect = titleRect;
590
591 painter.drawPixmap(rect.topLeft(), leftPm.mask());
592 painter.fillRect(rect.left() + leftPm.width(), rect.top(),
593 rect.width() - leftPm.width() - rightPm.width(),
594 centerPm.height(), Qt::color1);
595 painter.drawPixmap(rect.topRight() - QPoint(rightPm.width() - 1, 0), rightPm.mask());
596
597 painter.fillRect(contents, Qt::color1);
598 }
599
600 dw->setMask(bitmap);
601 }
