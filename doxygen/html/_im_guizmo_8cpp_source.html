104 void Frustum(float left, float right, float bottom, float top, float znear, float zfar, float* m16)
105 {
106 float temp, temp2, temp3, temp4;
107 temp = 2.0f * znear;
108 temp2 = right - left;
109 temp3 = top - bottom;
110 temp4 = zfar - znear;
111 m16[0] = temp / temp2;
112 m16[1] = 0.0;
113 m16[2] = 0.0;
114 m16[3] = 0.0;
115 m16[4] = 0.0;
116 m16[5] = temp / temp3;
117 m16[6] = 0.0;
118 m16[7] = 0.0;
119 m16[8] = (right + left) / temp2;
120 m16[9] = (top + bottom) / temp3;
121 m16[10] = (-zfar - znear) / temp4;
122 m16[11] = -1.0f;
123 m16[12] = 0.0;
124 m16[13] = 0.0;
125 m16[14] = (-temp * zfar) / temp4;
126 m16[15] = 0.0;
127 }

157 void LookAt(const float* eye, const float* at, const float* up, float* m16)
158 {
159 float X[3], Y[3], Z[3], tmp[3];
160
161 tmp[0] = eye[0] - at[0];
162 tmp[1] = eye[1] - at[1];
163 tmp[2] = eye[2] - at[2];
164 Normalize(tmp, Z);
165 Normalize(up, Y);
166 Cross(Y, Z, tmp);
167 Normalize(tmp, X);
168 Cross(Z, X, tmp);
169 Normalize(tmp, Y);
170
171 m16[0] = X[0];
172 m16[1] = Y[0];
173 m16[2] = Z[0];
174 m16[3] = 0.0f;
175 m16[4] = X[1];
176 m16[5] = Y[1];
177 m16[6] = Z[1];
178 m16[7] = 0.0f;
179 m16[8] = X[2];
180 m16[9] = Y[2];
181 m16[10] = Z[2];
182 m16[11] = 0.0f;
183 m16[12] = -Dot(X, eye);
184 m16[13] = -Dot(Y, eye);
185 m16[14] = -Dot(Z, eye);
186 m16[15] = 1.0f;
187 }

464 float matrix_t::Inverse(const matrix_t& srcMatrix, bool affine)
465 {
466 float det = 0;
467
468 if (affine)
469 {
470 det = GetDeterminant();
471 float s = 1 / det;
472 m[0][0] = (srcMatrix.m[1][1] * srcMatrix.m[2][2] - srcMatrix.m[1][2] * srcMatrix.m[2][1]) * s;
473 m[0][1] = (srcMatrix.m[2][1] * srcMatrix.m[0][2] - srcMatrix.m[2][2] * srcMatrix.m[0][1]) * s;
474 m[0][2] = (srcMatrix.m[0][1] * srcMatrix.m[1][2] - srcMatrix.m[0][2] * srcMatrix.m[1][1]) * s;
475 m[1][0] = (srcMatrix.m[1][2] * srcMatrix.m[2][0] - srcMatrix.m[1][0] * srcMatrix.m[2][2]) * s;
476 m[1][1] = (srcMatrix.m[2][2] * srcMatrix.m[0][0] - srcMatrix.m[2][0] * srcMatrix.m[0][2]) * s;
477 m[1][2] = (srcMatrix.m[0][2] * srcMatrix.m[1][0] - srcMatrix.m[0][0] * srcMatrix.m[1][2]) * s;
478 m[2][0] = (srcMatrix.m[1][0] * srcMatrix.m[2][1] - srcMatrix.m[1][1] * srcMatrix.m[2][0]) * s;
479 m[2][1] = (srcMatrix.m[2][0] * srcMatrix.m[0][1] - srcMatrix.m[2][1] * srcMatrix.m[0][0]) * s;
480 m[2][2] = (srcMatrix.m[0][0] * srcMatrix.m[1][1] - srcMatrix.m[0][1] * srcMatrix.m[1][0]) * s;
481 m[3][0] = -(m[0][0] * srcMatrix.m[3][0] + m[1][0] * srcMatrix.m[3][1] + m[2][0] * srcMatrix.m[3][2]);
482 m[3][1] = -(m[0][1] * srcMatrix.m[3][0] + m[1][1] * srcMatrix.m[3][1] + m[2][1] * srcMatrix.m[3][2]);
483 m[3][2] = -(m[0][2] * srcMatrix.m[3][0] + m[1][2] * srcMatrix.m[3][1] + m[2][2] * srcMatrix.m[3][2]);
484 }
485 else
486 {
487 // transpose matrix
488 float src[16];
489 for (int i = 0; i < 4; ++i)
490 {
491 src[i] = srcMatrix.m16[i * 4];
492 src[i + 4] = srcMatrix.m16[i * 4 + 1];
493 src[i + 8] = srcMatrix.m16[i * 4 + 2];
494 src[i + 12] = srcMatrix.m16[i * 4 + 3];
495 }
496
497 // calculate pairs for first 8 elements (cofactors)
498 float tmp[12]; // temp array for pairs
499 tmp[0] = src[10] * src[15];
500 tmp[1] = src[11] * src[14];
501 tmp[2] = src[9] * src[15];
502 tmp[3] = src[11] * src[13];
503 tmp[4] = src[9] * src[14];
504 tmp[5] = src[10] * src[13];
505 tmp[6] = src[8] * src[15];
506 tmp[7] = src[11] * src[12];
507 tmp[8] = src[8] * src[14];
508 tmp[9] = src[10] * src[12];
509 tmp[10] = src[8] * src[13];
510 tmp[11] = src[9] * src[12];
511
512 // calculate first 8 elements (cofactors)
513 m16[0] = (tmp[0] * src[5] + tmp[3] * src[6] + tmp[4] * src[7]) - (tmp[1] * src[5] + tmp[2] * src[6] + tmp[5] * src[7]);
514 m16[1] = (tmp[1] * src[4] + tmp[6] * src[6] + tmp[9] * src[7]) - (tmp[0] * src[4] + tmp[7] * src[6] + tmp[8] * src[7]);
515 m16[2] = (tmp[2] * src[4] + tmp[7] * src[5] + tmp[10] * src[7]) - (tmp[3] * src[4] + tmp[6] * src[5] + tmp[11] * src[7]);
516 m16[3] = (tmp[5] * src[4] + tmp[8] * src[5] + tmp[11] * src[6]) - (tmp[4] * src[4] + tmp[9] * src[5] + tmp[10] * src[6]);
517 m16[4] = (tmp[1] * src[1] + tmp[2] * src[2] + tmp[5] * src[3]) - (tmp[0] * src[1] + tmp[3] * src[2] + tmp[4] * src[3]);
518 m16[5] = (tmp[0] * src[0] + tmp[7] * src[2] + tmp[8] * src[3]) - (tmp[1] * src[0] + tmp[6] * src[2] + tmp[9] * src[3]);
519 m16[6] = (tmp[3] * src[0] + tmp[6] * src[1] + tmp[11] * src[3]) - (tmp[2] * src[0] + tmp[7] * src[1] + tmp[10] * src[3]);
520 m16[7] = (tmp[4] * src[0] + tmp[9] * src[1] + tmp[10] * src[2]) - (tmp[5] * src[0] + tmp[8] * src[1] + tmp[11] * src[2]);
521
522 // calculate pairs for second 8 elements (cofactors)
523 tmp[0] = src[2] * src[7];
524 tmp[1] = src[3] * src[6];
525 tmp[2] = src[1] * src[7];
526 tmp[3] = src[3] * src[5];
527 tmp[4] = src[1] * src[6];
528 tmp[5] = src[2] * src[5];
529 tmp[6] = src[0] * src[7];
530 tmp[7] = src[3] * src[4];
531 tmp[8] = src[0] * src[6];
532 tmp[9] = src[2] * src[4];
533 tmp[10] = src[0] * src[5];
534 tmp[11] = src[1] * src[4];
535
536 // calculate second 8 elements (cofactors)
537 m16[8] = (tmp[0] * src[13] + tmp[3] * src[14] + tmp[4] * src[15]) - (tmp[1] * src[13] + tmp[2] * src[14] + tmp[5] * src[15]);
538 m16[9] = (tmp[1] * src[12] + tmp[6] * src[14] + tmp[9] * src[15]) - (tmp[0] * src[12] + tmp[7] * src[14] + tmp[8] * src[15]);
539 m16[10] = (tmp[2] * src[12] + tmp[7] * src[13] + tmp[10] * src[15]) - (tmp[3] * src[12] + tmp[6] * src[13] + tmp[11] * src[15]);
540 m16[11] = (tmp[5] * src[12] + tmp[8] * src[13] + tmp[11] * src[14]) - (tmp[4] * src[12] + tmp[9] * src[13] + tmp[10] * src[14]);
541 m16[12] = (tmp[2] * src[10] + tmp[5] * src[11] + tmp[1] * src[9]) - (tmp[4] * src[11] + tmp[0] * src[9] + tmp[3] * src[10]);
542 m16[13] = (tmp[8] * src[11] + tmp[0] * src[8] + tmp[7] * src[10]) - (tmp[6] * src[10] + tmp[9] * src[11] + tmp[1] * src[8]);
543 m16[14] = (tmp[6] * src[9] + tmp[11] * src[11] + tmp[3] * src[8]) - (tmp[10] * src[11] + tmp[2] * src[8] + tmp[7] * src[9]);
544 m16[15] = (tmp[10] * src[10] + tmp[4] * src[8] + tmp[9] * src[9]) - (tmp[8] * src[9] + tmp[11] * src[10] + tmp[5] * src[8]);
545
546 // calculate determinant
547 det = src[0] * m16[0] + src[1] * m16[1] + src[2] * m16[2] + src[3] * m16[3];
548
549 // calculate matrix inverse
550 float invdet = 1 / det;
551 for (int j = 0; j < 16; ++j)
552 {
553 m16[j] *= invdet;
554 }
555 }
556
557 return det;
558 }

560 void matrix_t::RotationAxis(const vec_t& axis, float angle)
561 {
562 float length2 = axis.LengthSq();
563 if (length2 < FLT_EPSILON)
564 {
565 SetToIdentity();
566 return;
567 }
568
569 vec_t n = axis * (1.f / sqrtf(length2));
570 float s = sinf(angle);
571 float c = cosf(angle);
572 float k = 1.f - c;
573
574 float xx = n.x * n.x * k + c;
575 float yy = n.y * n.y * k + c;
576 float zz = n.z * n.z * k + c;
577 float xy = n.x * n.y * k;
578 float yz = n.y * n.z * k;
579 float zx = n.z * n.x * k;
580 float xs = n.x * s;
581 float ys = n.y * s;
582 float zs = n.z * s;
583
584 m[0][0] = xx;
585 m[0][1] = xy + zs;
586 m[0][2] = zx - ys;
587 m[0][3] = 0.f;
588 m[1][0] = xy - zs;
589 m[1][1] = yy;
590 m[1][2] = yz + xs;
591 m[1][3] = 0.f;
592 m[2][0] = zx + ys;
593 m[2][1] = yz - xs;
594 m[2][2] = zz;
595 m[2][3] = 0.f;
596 m[3][0] = 0.f;
597 m[3][1] = 0.f;
598 m[3][2] = 0.f;
599 m[3][3] = 1.f;
600 }

799 static float GetSegmentLengthClipSpace(const vec_t& start, const vec_t& end)
800 {
801 vec_t startOfSegment = start;
802 startOfSegment.TransformPoint(gContext.mMVP);
803 if (fabsf(startOfSegment.w) > FLT_EPSILON) // check for axis aligned with camera direction
804 {
805 startOfSegment *= 1.f / startOfSegment.w;

809 endOfSegment.TransformPoint(gContext.mMVP);

811 {
812 endOfSegment *= 1.f / endOfSegment.w;

814
815 vec_t clipSpaceAxis = endOfSegment - startOfSegment;
816 clipSpaceAxis.y /= gContext.mDisplayRatio;

843 inline vec_t PointOnSegment(const vec_t& point, const vec_t& vertPos1, const vec_t& vertPos2)
844 {
845 vec_t c = point - vertPos1;
846 vec_t V;
847
848 V.Normalize(vertPos2 - vertPos1);
849 float d = (vertPos2 - vertPos1).Length();
850 float t = V.Dot3(c);
851
852 if (t < 0.f)
853 {
854 return vertPos1;
855 }
856
857 if (t > d)
858 {
859 return vertPos2;
860 }
861
862 return vertPos1 + V * t;
863 }

1132 static void ComputeSnap(float* value, float snap)
1133 {
1134 if (snap <= FLT_EPSILON)
1135 {
1136 return;
1137 }
1138
1139 float modulo = fmodf(*value, snap);
1140 float moduloRatio = fabsf(modulo) / snap;
1141 if (moduloRatio < snapTension)
1142 {
1143 *value -= modulo;
1144 }
1145 else if (moduloRatio > (1.f - snapTension))
1146 {
1147 *value = *value - modulo + snap * ((*value < 0.f) ? -1.f : 1.f);
1148 }
1149 }
