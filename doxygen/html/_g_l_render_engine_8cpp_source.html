121 void GLRenderEngine::draw(dyno::SceneGraph* scene)
122 {
123 m_rparams.proj = mCamera->getProjMat();
124 m_rparams.view = mCamera->getViewMat();
125
126 // Graphscrene draw
127 GLint fbo;
128 glGetIntegerv(GL_FRAMEBUFFER_BINDING, &fbo);
129
130 // gather visual modules
131 RenderQueue renderQueue;
132 // enqueue render content
133 if (scene != nullptr && !scene->isEmpty())
134 {

136 }
137
138 // update shadow map
139 if(m_rparams.light.mainLightShadow != 0.f) {
140 mShadowMap->beginUpdate(scene, m_rparams);
141 renderQueue.draw(GLVisualModule::SHADOW);
142 mShadowMap->endUpdate();
143 }
144
145 // setup scene transform matrices
146 struct
147 {
148 glm::mat4 model;
149 glm::mat4 view;
150 glm::mat4 projection;
151 int width;
152 int height;
153 } sceneUniformBuffer;
154 sceneUniformBuffer.model = glm::mat4(1);
155 sceneUniformBuffer.view = m_rparams.view;
156 sceneUniformBuffer.projection = m_rparams.proj;
157 sceneUniformBuffer.width = m_rparams.viewport.w;
158 sceneUniformBuffer.height = m_rparams.viewport.h;
159
160 mTransformUBO.load(&sceneUniformBuffer, sizeof(sceneUniformBuffer));
161 mTransformUBO.bindBufferBase(0);
162
163 // setup light block
164 RenderParams::Light light = m_rparams.light;
165 light.mainLightDirection = glm::vec3(m_rparams.view * glm::vec4(light.mainLightDirection, 0));
166 mLightUBO.load(&light, sizeof(light));
167 mLightUBO.bindBufferBase(1);
168
169 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, fbo);
170 glViewport(0, 0, m_rparams.viewport.w, m_rparams.viewport.h);
171
172 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
173
174 Vec3f c0 = Vec3f(m_rparams.bgColor0.x, m_rparams.bgColor0.y, m_rparams.bgColor0.z);
175 Vec3f c1 = Vec3f(m_rparams.bgColor1.x, m_rparams.bgColor1.y, m_rparams.bgColor1.z);
176 mRenderHelper->drawBackground(c0, c1);
177
178 glClear(GL_DEPTH_BUFFER_BIT);
179
180 // render modules
181 renderQueue.draw(GLVisualModule::COLOR);
182
183 // draw a plane
184 if (m_rparams.showGround)
185 {
186 mRenderHelper->drawGround(m_rparams.planeScale* mCamera->distanceUnit(), m_rparams.rulerScale* mCamera->distanceUnit());
187 }
188
189 // draw scene bounding box
190 if (m_rparams.showSceneBounds && scene != 0)
191 {
192 // get bounding box of the scene
193 auto p0 = scene->getLowerBound();
194 auto p1 = scene->getUpperBound();

196 }
197 // draw axis
198 if (m_rparams.showAxisHelper)
