181 void GLRenderEngine::updateRenderModules(dyno::SceneGraph* scene)
182 {
183
184 std::vector<RenderItem> items;
185
186 for (auto iter = scene->begin(); iter != scene->end(); iter++) {
187 auto node = iter.get();
188
189 for (auto m : node->graphicsPipeline()->activeModules()) {
190 auto vm = std::dynamic_pointer_cast<GLVisualModule>(m);
191 if (vm) {
192 items.push_back({ node, vm });
193 }
194 }
195 }
196
197
198 for (auto item : mRenderItems) {
199 // release GL resource for unreferenced visual module
200 if (std::find(items.begin(), items.end(), item) == items.end())
201 {
202 item.visualModule->destroyGL();
203 }
204 }
205
206 mRenderItems = items;
207
208
209 // initialize modules
210 for (auto item : mRenderItems) {
211 if (!item.visualModule->isGLInitialized)
212 item.visualModule->isGLInitialized = item.visualModule->initializeGL();
213
214 if (!item.visualModule->isGLInitialized)
215 printf("Warning: failed to initialize %s\n", item.visualModule->getName().c_str());
216 }
217
218 // update if necessary
219 for (auto item : mRenderItems) {
220 if (item.visualModule->isGLInitialized)
221 {
222 // check update
223 if (item.visualModule->changed > item.visualModule->updated) {
224 item.visualModule->updateGL();
225 item.visualModule->updated = GLVisualModule::clock::now();
226 }
227 }

400 {
401 // TODO: check valid input
402 w = std::max(1, w);
403 h = std::max(1, h);
404
405 // read pixels
406 std::vector<glm::ivec4> indices(w * h);
407
408 mFramebuffer.bind(GL_READ_FRAMEBUFFER);
409 glReadBuffer(GL_COLOR_ATTACHMENT1);
410 //glPixelStorei(GL_PACK_ALIGNMENT, 1);
411 glReadPixels(x, y, w, h, GL_RGBA_INTEGER, GL_INT, indices.data());
412 gl::glCheckError();
413
414 // use unordered set to get unique id
415 std::unordered_set<glm::ivec4> uniqueIdx(indices.begin(), indices.end());
416
417 Selection result;
418 result.x = x;
419 result.y = y;
420 result.w = w;
421 result.h = h;
422
423 for (const auto& idx : uniqueIdx) {
424 const int nodeIdx = idx.x;
425 const int instIdx = idx.y;
426 const int primIdx = idx.z;
427
428 if (nodeIdx >= 0 && nodeIdx < mRenderItems.size()) {
429 result.items.push_back({mRenderItems[nodeIdx].node, instIdx, primIdx});
430 }
431 }
432
433 return result;
434 }
